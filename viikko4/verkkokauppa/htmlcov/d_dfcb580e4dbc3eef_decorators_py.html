697class lazyproperty(property): 

698 """ 

699 Works similarly to property(), but computes the value only once. 

700 

701 This essentially memorizes the value of the property by storing the result 

702 of its computation in the ``__dict__`` of the object instance. This is 

703 useful for computing the value of some property that should otherwise be 

704 invariant. For example:: 

705 

706 >>> class LazyTest: 

707 ... @lazyproperty 

708 ... def complicated_property(self): 

709 ... print('Computing the value for complicated_property...') 

710 ... return 42 

711 ... 

712 >>> lt = LazyTest() 

713 >>> lt.complicated_property 

714 Computing the value for complicated_property... 

715 42 

716 >>> lt.complicated_property 

717 42 

718 

719 As the example shows, the second time ``complicated_property`` is accessed, 

720 the ``print`` statement is not executed. Only the return value from the 

721 first access off ``complicated_property`` is returned. 

722 

723 By default, a setter and deleter are used which simply overwrite and 

724 delete, respectively, the value stored in ``__dict__``. Any user-specified 

725 setter or deleter is executed before executing these default actions. 

726 The one exception is that the default setter is not run if the user setter 

727 already sets the new value in ``__dict__`` and returns that value and the 

728 returned value is not ``None``. 

729 

730 Adapted from the recipe at 

731 http://code.activestate.com/recipes/363602-lazy-property-evaluation 

732 """ 

733 

734 def __init__(self, fget, fset=None, fdel=None, doc=None): 

735 super().__init__(fget, fset, fdel, doc) 

736 self._key = self.fget.__name__ 

737 

738 def __get__(self, obj, owner=None): 

739 try: 

740 val = obj.__dict__.get(self._key, _NotFound) 

741 if val is not _NotFound: 

742 return val 

743 else: 

744 val = self.fget(obj) 

745 obj.__dict__[self._key] = val 

746 return val 

747 except AttributeError: 

748 if obj is None: 

749 return self 

750 raise 

751 

752 def __set__(self, obj, val): 

753 obj_dict = obj.__dict__ 

754 if self.fset: 

755 ret = self.fset(obj, val) 

756 if ret is not None and obj_dict.get(self._key) is ret: 

757 # By returning the value set the setter signals that it took 

758 # over setting the value in obj.__dict__; this mechanism allows 

759 # it to override the input value 

760 return 

761 obj_dict[self._key] = val 

762 

763 def __delete__(self, obj): 

764 if self.fdel: 

765 self.fdel(obj) 

766 if self._key in obj.__dict__: 

767 del obj.__dict__[self._key]

770class sharedmethod(classmethod): 

771 """ 

772 This is a method decorator that allows both an instancemethod and a 

773 `classmethod` to share the same name. 

774 

775 When using `sharedmethod` on a method defined in a class's body, it 

776 may be called on an instance, or on a class. In the former case it 

777 behaves like a normal instance method (a reference to the instance is 

778 automatically passed as the first ``self`` argument of the method):: 

779 

780 >>> class Example: 

781 ... @sharedmethod 

782 ... def identify(self, *args): 

783 ... print('self was', self) 

784 ... print('additional args were', args) 

785 ... 

786 >>> ex = Example() 

787 >>> ex.identify(1, 2) 

788 self was <astropy.utils.decorators.Example object at 0x...> 

789 additional args were (1, 2) 

790 

791 In the latter case, when the `sharedmethod` is called directly from a 

792 class, it behaves like a `classmethod`:: 

793 

794 >>> Example.identify(3, 4) 

795 self was <class 'astropy.utils.decorators.Example'> 

796 additional args were (3, 4) 

797 

798 This also supports a more advanced usage, where the `classmethod` 

799 implementation can be written separately. If the class's *metaclass* 

800 has a method of the same name as the `sharedmethod`, the version on 

801 the metaclass is delegated to:: 

802 

803 >>> class ExampleMeta(type): 

804 ... def identify(self): 

805 ... print('this implements the {0}.identify ' 

806 ... 'classmethod'.format(self.__name__)) 

807 ... 

808 >>> class Example(metaclass=ExampleMeta): 

809 ... @sharedmethod 

810 ... def identify(self): 

811 ... print('this implements the instancemethod') 

812 ... 

813 >>> Example().identify() 

814 this implements the instancemethod 

815 >>> Example.identify() 

816 this implements the Example.identify classmethod 

817 """ 

818 

819 def __get__(self, obj, objtype=None): 

820 if obj is None: 

821 mcls = type(objtype) 

822 clsmeth = getattr(mcls, self.__func__.__name__, None) 

823 if callable(clsmeth): 

824 func = clsmeth 

825 else: 

826 func = self.__func__ 

827 

828 return self._make_method(func, objtype) 

829 else: 

830 return self._make_method(self.__func__, obj) 

831 

832 @staticmethod 

833 def _make_method(func, instance): 

834 return types.MethodType(func, instance)

837def format_doc(docstring, *args, **kwargs): 

838 """ 

839 Replaces the docstring of the decorated object and then formats it. 

840 

841 The formatting works like :meth:`str.format` and if the decorated object 

842 already has a docstring this docstring can be included in the new 

843 documentation if you use the ``{__doc__}`` placeholder. 

844 Its primary use is for reusing a *long* docstring in multiple functions 

845 when it is the same or only slightly different between them. 

846 

847 Parameters 

848 ---------- 

849 docstring : str or object or None 

850 The docstring that will replace the docstring of the decorated 

851 object. If it is an object like a function or class it will 

852 take the docstring of this object. If it is a string it will use the 

853 string itself. One special case is if the string is ``None`` then 

854 it will use the decorated functions docstring and formats it. 

855 

856 args : 

857 passed to :meth:`str.format`. 

858 

859 kwargs : 

860 passed to :meth:`str.format`. If the function has a (not empty) 

861 docstring the original docstring is added to the kwargs with the 

862 keyword ``'__doc__'``. 

863 

864 Raises 

865 ------ 

866 ValueError 

867 If the ``docstring`` (or interpreted docstring if it was ``None`` 

868 or not a string) is empty. 

869 

870 IndexError, KeyError 

871 If a placeholder in the (interpreted) ``docstring`` was not filled. see 

872 :meth:`str.format` for more information. 

873 

874 Notes 

875 ----- 

876 Using this decorator allows, for example Sphinx, to parse the 

877 correct docstring. 

878 

879 Examples 

880 -------- 

881 

882 Replacing the current docstring is very easy:: 

883 

884 >>> from astropy.utils.decorators import format_doc 

885 >>> @format_doc('''Perform num1 + num2''') 

886 ... def add(num1, num2): 

887 ... return num1+num2 

888 ... 

889 >>> help(add) # doctest: +SKIP 

890 Help on function add in module __main__: 

891 <BLANKLINE> 

892 add(num1, num2) 

893 Perform num1 + num2 

894 

895 sometimes instead of replacing you only want to add to it:: 

896 

897 >>> doc = ''' 

898 ... {__doc__} 

899 ... Parameters 

900 ... ---------- 

901 ... num1, num2 : Numbers 

902 ... Returns 

903 ... ------- 

904 ... result: Number 

905 ... ''' 

906 >>> @format_doc(doc) 

907 ... def add(num1, num2): 

908 ... '''Perform addition.''' 

909 ... return num1+num2 

910 ... 

911 >>> help(add) # doctest: +SKIP 

912 Help on function add in module __main__: 

913 <BLANKLINE> 

914 add(num1, num2) 

915 Perform addition. 

916 Parameters 

917 ---------- 

918 num1, num2 : Numbers 

919 Returns 

920 ------- 

921 result : Number 

922 

923 in case one might want to format it further:: 

924 

925 >>> doc = ''' 

926 ... Perform {0}. 

927 ... Parameters 

928 ... ---------- 

929 ... num1, num2 : Numbers 

930 ... Returns 

931 ... ------- 

932 ... result: Number 

933 ... result of num1 {op} num2 

934 ... {__doc__} 

935 ... ''' 

936 >>> @format_doc(doc, 'addition', op='+') 

937 ... def add(num1, num2): 

938 ... return num1+num2 

939 ... 

940 >>> @format_doc(doc, 'subtraction', op='-') 

941 ... def subtract(num1, num2): 

942 ... '''Notes: This one has additional notes.''' 

943 ... return num1-num2 

944 ... 

945 >>> help(add) # doctest: +SKIP 

946 Help on function add in module __main__: 

947 <BLANKLINE> 

948 add(num1, num2) 

949 Perform addition. 

950 Parameters 

951 ---------- 

952 num1, num2 : Numbers 

953 Returns 

954 ------- 

955 result : Number 

956 result of num1 + num2 

957 >>> help(subtract) # doctest: +SKIP 

958 Help on function subtract in module __main__: 

959 <BLANKLINE> 

960 subtract(num1, num2) 

961 Perform subtraction. 

962 Parameters 

963 ---------- 

964 num1, num2 : Numbers 

965 Returns 

966 ------- 

967 result : Number 

968 result of num1 - num2 

969 Notes : This one has additional notes. 

970 

971 These methods can be combined an even taking the docstring from another 

972 object is possible as docstring attribute. You just have to specify the 

973 object:: 

974 

975 >>> @format_doc(add) 

976 ... def another_add(num1, num2): 

977 ... return num1 + num2 

978 ... 

979 >>> help(another_add) # doctest: +SKIP 

980 Help on function another_add in module __main__: 

981 <BLANKLINE> 

982 another_add(num1, num2) 

983 Perform addition. 

984 Parameters 

985 ---------- 

986 num1, num2 : Numbers 

987 Returns 

988 ------- 

989 result : Number 

990 result of num1 + num2 

991 

992 But be aware that this decorator *only* formats the given docstring not 

993 the strings passed as ``args`` or ``kwargs`` (not even the original 

994 docstring):: 

995 

996 >>> @format_doc(doc, 'addition', op='+') 

997 ... def yet_another_add(num1, num2): 

998 ... '''This one is good for {0}.''' 

999 ... return num1 + num2 

1000 ... 

1001 >>> help(yet_another_add) # doctest: +SKIP 

1002 Help on function yet_another_add in module __main__: 

1003 <BLANKLINE> 

1004 yet_another_add(num1, num2) 

1005 Perform addition. 

1006 Parameters 

1007 ---------- 

1008 num1, num2 : Numbers 

1009 Returns 

1010 ------- 

1011 result : Number 

1012 result of num1 + num2 

1013 This one is good for {0}. 

1014 

1015 To work around it you could specify the docstring to be ``None``:: 

1016 

1017 >>> @format_doc(None, 'addition') 

1018 ... def last_add_i_swear(num1, num2): 

1019 ... '''This one is good for {0}.''' 

1020 ... return num1 + num2 

1021 ... 

1022 >>> help(last_add_i_swear) # doctest: +SKIP 

1023 Help on function last_add_i_swear in module __main__: 

1024 <BLANKLINE> 

1025 last_add_i_swear(num1, num2) 

1026 This one is good for addition. 

1027 

1028 Using it with ``None`` as docstring allows to use the decorator twice 

1029 on an object to first parse the new docstring and then to parse the 

1030 original docstring or the ``args`` and ``kwargs``. 

1031 """ 

1032 def set_docstring(obj): 

1033 if docstring is None: 

1034 # None means: use the objects __doc__ 

1035 doc = obj.__doc__ 

1036 # Delete documentation in this case so we don't end up with 

1037 # awkwardly self-inserted docs. 

1038 obj.__doc__ = None 

1039 elif isinstance(docstring, str): 

1040 # String: use the string that was given 

1041 doc = docstring 

1042 else: 

1043 # Something else: Use the __doc__ of this 

1044 doc = docstring.__doc__ 

1045 

1046 if not doc: 

1047 # In case the docstring is empty it's probably not what was wanted. 

1048 raise ValueError('docstring must be a string or containing a ' 

1049 'docstring that is not empty.') 

1050 

1051 # If the original has a not-empty docstring append it to the format 

1052 # kwargs. 

1053 kwargs['__doc__'] = obj.__doc__ or '' 

1054 obj.__doc__ = doc.format(*args, **kwargs) 

1055 return obj 

1056 return set_docstring
