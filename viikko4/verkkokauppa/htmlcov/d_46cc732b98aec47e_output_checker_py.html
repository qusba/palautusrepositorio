34class OutputChecker(doctest.OutputChecker): 

35 """ 

36 - Removes u'' prefixes on string literals 

37 - Ignores the 'L' suffix on long integers 

38 - In Numpy dtype strings, removes the leading pipe, i.e. '|S9' -> 

39 'S9'. Numpy 1.7 no longer includes it in display. 

40 - Supports the FLOAT_CMP flag, which parses floating point values 

41 out of the output and compares their numerical values rather than their 

42 string representation. This naturally supports complex numbers as well 

43 (simply by comparing their real and imaginary parts separately). 

44 """ 

45 rtol = 1e-05 

46 atol = 1e-08 

47 

48 _original_output_checker = doctest.OutputChecker 

49 

50 _str_literal_re = re.compile( 

51 r"(\W|^)[uU]([rR]?[\'\"])", re.UNICODE) 

52 _byteorder_re = re.compile( 

53 r"([\'\"])[|<>]([biufcSaUV][0-9]+)([\'\"])", re.UNICODE) 

54 _fix_32bit_re = re.compile( 

55 r"([\'\"])([iu])[48]([\'\"])", re.UNICODE) 

56 _long_int_re = re.compile( 

57 r"([0-9]+)L", re.UNICODE) 

58 

59 def __init__(self): 

60 # NOTE OutputChecker is an old-style class with no __init__ method, 

61 # so we can't call the base class version of __init__ here 

62 

63 exp = r'(?:e[+-]?\d+)' 

64 

65 got_floats = (r'\s*([+-]?\d+\.\d*{0}?|' 

66 r'[+-]?\.\d+{0}?|' 

67 r'[+-]?\d+{0}|' 

68 r'nan|' 

69 r'[+-]?inf)').format(exp) 

70 

71 # floats in the 'want' string may contain ellipses 

72 want_floats = got_floats + r'(\.{3})?' 

73 

74 front_sep = r'\s|[*+-,<=(\[]' 

75 back_sep = front_sep + r'|[>j)\]]' 

76 

77 fbeg = r'^{}(?={}|$)'.format(got_floats, back_sep) 

78 fmidend = r'(?<={}){}(?={}|$)'.format(front_sep, got_floats, back_sep) 

79 self.num_got_rgx = re.compile(r'({}|{})'.format(fbeg, fmidend)) 

80 

81 fbeg = r'^{}(?={}|$)'.format(want_floats, back_sep) 

82 fmidend = r'(?<={}){}(?={}|$)'.format(front_sep, want_floats, back_sep) 

83 self.num_want_rgx = re.compile(r'({}|{})'.format(fbeg, fmidend)) 

84 

85 def do_fixes(self, want, got): 

86 want = re.sub(self._str_literal_re, r'\1\2', want) 

87 want = re.sub(self._byteorder_re, r'\1\2\3', want) 

88 want = re.sub(self._fix_32bit_re, r'\1\2\3', want) 

89 want = re.sub(self._long_int_re, r'\1', want) 

90 

91 got = re.sub(self._str_literal_re, r'\1\2', got) 

92 got = re.sub(self._byteorder_re, r'\1\2\3', got) 

93 got = re.sub(self._fix_32bit_re, r'\1\2\3', got) 

94 got = re.sub(self._long_int_re, r'\1', got) 

95 

96 return want, got

98 def find_numbers(self, text): 

99 """ 

100 Find float strings in text. 

101 >>> OutputChecker().find_numbers("1.1 foo abr 2.22") 

102 ['1.1', '2.22'] 

103 """ 

104 matches = self.num_want_rgx.finditer(text) 

105 return [match.group(1) for match in matches]

107 def equal_floats(self, a, b): 

108 """ 

109 Compare float strings. 

110 >>> OutputChecker().equal_floats('1.1', '1.10000000001') 

111 True 

112 >>> OutputChecker().equal_floats('1.1', '1.11') 

113 False 

114 """ 

115 a, b = float(a), float(b) 

116 return isclose(a, b, rtol=self.rtol, atol=self.atol)

118 def startswith(self, arr, prefix): 

119 """ 

120 Check if array of str/floats starts with floats in prefix. 

121 >>> OutputChecker().startswith(['1', '2', '3'], ['1', '2.00000000001']) 

122 True 

123 >>> OutputChecker().startswith(['1', '2', '3'], ['1', '2.1']) 

124 False 

125 """ 

126 if len(prefix) == 0: 

127 return True 

128 if len(arr) < len(prefix): 

129 return False 

130 for a, b in zip(arr, prefix): 

131 if not self.equal_floats(a, b): 

132 return False 

133 return True 

134 

135 def endswith(self, arr, postfix): 

136 """ 

137 Check if array of str/floats ends with floats in postfix. 

138 >>> OutputChecker().endswith(['1', '2', '3'], ['2', '3.00000000001']) 

139 True 

140 >>> OutputChecker().endswith(['1', '2', '3'], ['2', '3.1']) 

141 False 

142 """ 

143 return self.startswith(arr[::-1], postfix[::-1])

145 def find(self, arr, suffix, start, end): 

146 """ 

147 Search for floats from suffix in arr. 

148 >>> OutputChecker().find(['1', '2', '3', '4'], ['2', '3.00000000001'], 0, 4) 

149 1 

150 >>> OutputChecker().find(['1', '2', '3', '4'], ['2', '3.1'], 0, 4) 

151 -1 

152 """ 

153 if len(suffix) == 0: 

154 return start 

155 arr = arr[start:end] 

156 for i, a in enumerate(arr): 

157 # if current floats match... 

158 if self.equal_floats(a, suffix[0]): 

159 # ... then compare the rest of numbers from suffix 

160 if self.startswith(arr[i:], suffix): 

161 return start + i 

162 return -1

164 def partial_match(self, arr, chunks): 

165 """ 

166 Check that each chunk in chunks is inside provided array of strings/floats. 

167 This is essentially list-with-floats equivalent of ellipsis matching. 

168 >>> OutputChecker().partial_match( 

169 ... ['1', '2', '3', '4'], 

170 ... [['1', '2'], ['4']], 

171 ... ) 

172 True 

173 >>> OutputChecker().partial_match( 

174 ... ['1', '2', '3', '4'], 

175 ... [['1', '2'], []], 

176 ... ) 

177 True 

178 >>> OutputChecker().partial_match( 

179 ... ['1', '2', '3', '4'], 

180 ... [['1', '2'], ['5']], 

181 ... ) 

182 False 

183 """ 

184 assert len(chunks) >= 2 

185 startpos, endpos = 0, len(arr) 

186 chunk = chunks[0] 

187 if chunk: # starts with exact match 

188 if self.startswith(arr, chunk): 

189 startpos = len(chunk) 

190 del chunks[0] 

191 else: 

192 return False 

193 chunk = chunks[-1] 

194 if chunk: # ends with exact match 

195 if self.endswith(arr, chunk): 

196 endpos -= len(chunk) 

197 del chunks[-1] 

198 else: 

199 return False 

200 

201 if startpos > endpos: 

202 return False 

203 

204 for chunk in chunks: 

205 startpos = self.find(arr, chunk, startpos, endpos) 

206 if startpos < 0: 

207 return False 

208 startpos += len(chunk) 

209 

210 return True

212 def normalize_floats(self, want, got, flags): 

213 """ 

214 Alternative to the built-in check_output that also handles parsing 

215 float values and comparing their numeric values rather than their 

216 string representations. 

217 

218 This requires rewriting enough of the basic check_output that, when 

219 FLOAT_CMP is enabled, it totally takes over for check_output. 

220 """ 

221 

222 # <BLANKLINE> can be used as a special sequence to signify a 

223 # blank line, unless the DONT_ACCEPT_BLANKLINE flag is used. 

224 if not (flags & doctest.DONT_ACCEPT_BLANKLINE): 

225 # Replace <BLANKLINE> in want with a blank line. 

226 want = re.sub(r'(?m)^{}\s*?$'.format(re.escape(doctest.BLANKLINE_MARKER)), 

227 '', want) 

228 # If a line in got contains only spaces, then remove the 

229 # spaces. 

230 got = re.sub(r'(?m)^\s*?$', '', got) 

231 

232 # This flag causes doctest to ignore any differences in the 

233 # contents of whitespace strings. Note that this can be used 

234 # in conjunction with the ELLIPSIS flag. 

235 if flags & doctest.NORMALIZE_WHITESPACE: 

236 got = ' '.join(got.split()) 

237 want = ' '.join(want.split()) 

238 

239 # Handle the common case first, for efficiency: 

240 # if they're string-identical, always return true. 

241 if got == want: 

242 return True 

243 

244 got_ = self.num_got_rgx.sub('0.0', got) 

245 want_ = self.num_got_rgx.sub('0.0', want) 

246 # fail if strings with ellipsis and normalize floats are not equal 

247 if flags & doctest.ELLIPSIS: 

248 if not doctest._ellipsis_match(want_, got_): 

249 return False 

250 else: 

251 if not got_ == want_: 

252 return False 

253 

254 # at this point we made sure that non-float parts of strings are equivalent 

255 # so now we need to compare each number 

256 

257 numbers_got = self.find_numbers(got) 

258 numbers_want_chunks = [ 

259 self.find_numbers(chunk) 

260 for chunk in want.split(doctest.ELLIPSIS_MARKER) 

261 ] 

262 if flags & doctest.ELLIPSIS and len(numbers_want_chunks) >= 2: 

263 return self.partial_match(numbers_got, numbers_want_chunks) 

264 

265 # TODO parse integers as well ? 

266 # Parse floats and compare them. 

267 numbers_want = [f for chunk in numbers_want_chunks for f in chunk] # flatten array 

268 if len(numbers_got) != len(numbers_want): 

269 return False 

270 for ng, nw in zip(numbers_got, numbers_want): 

271 if not self.equal_floats(ng, nw): 

272 return False 

273 

274 return True

276 def check_output(self, want, got, flags): 

277 if (flags & IGNORE_OUTPUT or (six.PY2 and flags & IGNORE_OUTPUT_2) or 

278 (not six.PY2 and flags & IGNORE_OUTPUT_3)): 

279 return True 

280 

281 if flags & FIX: 

282 want, got = self.do_fixes(want, got) 

283 

284 if flags & FLOAT_CMP: 

285 return self.normalize_floats(want, got, flags) 

286

288 # new-style class. 

289 return self._original_output_checker.check_output( 

290 self, want, got, flags)
