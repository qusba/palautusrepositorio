64def integer_range(data, lower, upper, center=None): 

65 assert lower <= upper 

66 if lower == upper: 

67 # Write a value even when this is trival so that when a bound depends 

68 # on other values we don't suddenly disappear when the gap shrinks to 

69 # zero - if that happens then often the data stream becomes misaligned 

70 # and we fail to shrink in cases where we really should be able to. 

71 data.draw_bits(1, forced=0) 

72 return int(lower) 

73 

74 if center is None: 

75 center = lower 

76 center = min(max(center, lower), upper) 

77 

78 if center == upper: 

79 above = False 

80 elif center == lower: 

81 above = True 

82 else: 

83 above = boolean(data) 

84 

85 if above: 

86 gap = upper - center 

87 else: 

88 gap = center - lower 

89 

90 assert gap > 0 

91 

92 bits = bit_length(gap) 

93 probe = gap + 1 

94 

95 if bits > 24 and data.draw_bits(3): 

96 # For large ranges, we combine the uniform random distribution from draw_bits 

97 # with the weighting scheme used by WideRangeIntStrategy with moderate chance. 

98 # Cutoff at 2 ** 24 so unicode choice is uniform but 32bit distribution is not. 

99 idx = Sampler([4.0, 8.0, 1.0, 1.0, 0.5]).sample(data) 

100 sizes = [8, 16, 32, 64, 128] 

101 bits = min(bits, sizes[idx]) 

102 

103 while probe > gap: 

104 data.start_example(INTEGER_RANGE_DRAW_LABEL) 

105 probe = data.draw_bits(bits) 

106 data.stop_example(discard=probe > gap) 

107 

108 if above: 

109 result = center + probe 

110 else: 

111 result = center - probe 

112 

113 assert lower <= result <= upper 

114 return int(result)

169def biased_coin(data, p): 

170 """Return False with probability p (assuming a uniform generator), 

171 shrinking towards False.""" 

172 data.start_example(BIASED_COIN_LABEL) 

173 while True: 

174 # The logic here is a bit complicated and special cased to make it 

175 # play better with the shrinker. 

176 

177 # We imagine partitioning the real interval [0, 1] into 256 equal parts 

178 # and looking at each part and whether its interior is wholly <= p 

179 # or wholly >= p. At most one part can be neither. 

180 

181 # We then pick a random part. If it's wholly on one side or the other 

182 # of p then we use that as the answer. If p is contained in the 

183 # interval then we start again with a new probability that is given 

184 # by the fraction of that interval that was <= our previous p. 

185 

186 # We then take advantage of the fact that we have control of the 

187 # labelling to make this shrink better, using the following tricks: 

188 

189 # If p is <= 0 or >= 1 the result of this coin is certain. We make sure 

190 # to write a byte to the data stream anyway so that these don't cause 

191 # difficulties when shrinking. 

192 if p <= 0: 

193 data.draw_bits(1, forced=0) 

194 result = False 

195 elif p >= 1: 

196 data.draw_bits(1, forced=1) 

197 result = True 

198 else: 

199 falsey = floor(256 * (1 - p)) 

200 truthy = floor(256 * p) 

201 remainder = 256 * p - truthy 

202 

203 if falsey + truthy == 256: 

204 if isinstance(p, Fraction): 

205 m = p.numerator 

206 n = p.denominator 

207 else: 

208 m, n = p.as_integer_ratio() 

209 assert n & (n - 1) == 0, n # n is a power of 2 

210 assert n > m > 0 

211 truthy = m 

212 falsey = n - m 

213 bits = bit_length(n) - 1 

214 partial = False 

215 else: 

216 bits = 8 

217 partial = True 

218 

219 i = data.draw_bits(bits) 

220 

221 # We always label the region that causes us to repeat the loop as 

222 # 255 so that shrinking this byte never causes us to need to draw 

223 # more data. 

224 if partial and i == 255: 

225 p = remainder 

226 continue 

227 if falsey == 0: 

228 # Every other partition is truthy, so the result is true 

229 result = True 

230 elif truthy == 0: 

231 # Every other partition is falsey, so the result is false 

232 result = False 

233 elif i <= 1: 

234 # We special case so that zero is always false and 1 is always 

235 # true which makes shrinking easier because we can always 

236 # replace a truthy block with 1. This has the slightly weird 

237 # property that shrinking from 2 to 1 can cause the result to 

238 # grow, but the shrinker always tries 0 and 1 first anyway, so 

239 # this will usually be fine. 

240 result = bool(i) 

241 else: 

242 # Originally everything in the region 0 <= i < falsey was false 

243 # and everything above was true. We swapped one truthy element 

244 # into this region, so the region becomes 0 <= i <= falsey 

245 # except for i = 1. We know i > 1 here, so the test for truth 

246 # becomes i > falsey. 

247 result = i > falsey 

248 break 

249 data.stop_example() 

250 return result

253class Sampler(object): 

254 """Sampler based on Vose's algorithm for the alias method. See 

255 http://www.keithschwarz.com/darts-dice-coins/ for a good explanation. 

256 

257 The general idea is that we store a table of triples (base, alternate, p). 

258 base. We then pick a triple uniformly at random, and choose its alternate 

259 value with probability p and else choose its base value. The triples are 

260 chosen so that the resulting mixture has the right distribution. 

261 

262 We maintain the following invariants to try to produce good shrinks: 

263 

264 1. The table is in lexicographic (base, alternate) order, so that choosing 

265 an earlier value in the list always lowers (or at least leaves 

266 unchanged) the value. 

267 2. base[i] < alternate[i], so that shrinking the draw always results in 

268 shrinking the chosen element. 

269 """

285 large = []

288 scaled_probabilities = [] 

289

291 scaled = p * n 

292 scaled_probabilities.append(scaled)

294 self.table[i][2] = zero 

295 elif scaled < 1:

297 else:

302 while small and large:

312 scaled_probabilities[hi] + scaled_probabilities[lo]

331 entry[0], entry[1] = entry[1], entry[0]

333 self.table.sort()

337 i = integer_range(data, 0, len(self.table) - 1)

347class many(object): 

348 """Utility class for collections. Bundles up the logic we use for "should I 

349 keep drawing more values?" and handles starting and stopping examples in 

350 the right place. 

351 

352 Intended usage is something like: 

353 

354 elements = many(data, ...) 

355 while elements.more(): 

356 add_stuff_to_result() 

357 """ 

358 

359 def __init__(self, data, min_size, max_size, average_size): 

360 assert 0 <= min_size <= average_size <= max_size 

361 self.min_size = min_size 

362 self.max_size = max_size 

363 self.data = data 

364 self.stopping_value = 1 - 1.0 / (1 + average_size) 

365 self.count = 0 

366 self.rejections = 0 

367 self.drawn = False 

368 self.force_stop = False 

369 self.rejected = False 

370 

371 def more(self): 

372 """Should I draw another element to add to the collection?""" 

373 if self.drawn: 

374 self.data.stop_example(discard=self.rejected) 

375 

376 self.drawn = True 

377 self.rejected = False 

378 

379 self.data.start_example(ONE_FROM_MANY_LABEL) 

380 

381 if self.min_size == self.max_size: 

382 should_continue = self.count < self.min_size 

383 elif self.force_stop: 

384 should_continue = False 

385 else: 

386 if self.count < self.min_size: 

387 p_continue = 1.0 

388 elif self.count >= self.max_size: 

389 p_continue = 0.0 

390 else: 

391 p_continue = self.stopping_value 

392 should_continue = biased_coin(self.data, p_continue) 

393 

394 if should_continue: 

395 self.count += 1 

396 return True 

397 else: 

398 self.data.stop_example() 

399 return False 

400 

401 def reject(self): 

402 """Reject the last example (i.e. don't count it towards our budget of 

403 elements because it's not going to go in the final collection).""" 

404 assert self.count > 0 

405 self.count -= 1 

406 self.rejections += 1 

407 self.rejected = True 

408 if self.rejections > 2 * self.count: 

409 if self.count < self.min_size: 

410 self.data.mark_invalid() 

411 else: 

412 self.force_stop = True
