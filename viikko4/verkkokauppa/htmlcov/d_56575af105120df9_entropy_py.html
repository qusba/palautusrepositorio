44def register_random(r): 

45 # type: (random.Random) -> None 

46 """Register the given Random instance for management by Hypothesis. 

47 

48 You can pass ``random.Random`` instances (or other objects with seed, 

49 getstate, and setstate methods) to ``register_random(r)`` to have their 

50 states seeded and restored in the same way as the global PRNGs from the 

51 ``random`` and ``numpy.random`` modules. 

52 

53 All global PRNGs, from e.g. simulation or scheduling frameworks, should 

54 be registered to prevent flaky tests. Hypothesis will ensure that the 

55 PRNG state is consistent for all test runs, or reproducibly varied if you 

56 choose to use the :func:`~hypothesis.strategies.random_module` strategy. 

57 """ 

58 if not (hasattr(r, "seed") and hasattr(r, "getstate") and hasattr(r, "setstate")): 

59 raise InvalidArgument("r=%r does not have all the required methods" % (r,)) 

60 if r not in RANDOMS_TO_MANAGE: 

61 RANDOMS_TO_MANAGE.append(r)

64def get_seeder_and_restorer(seed=0): 

65 """Return a pair of functions which respectively seed all and restore 

66 the state of all registered PRNGs. 

67 

68 This is used by the core engine via `deterministic_PRNG`, and by users 

69 via `register_random`. We support registration of additional random.Random 

70 instances (or other objects with seed, getstate, and setstate methods) 

71 to force determinism on simulation or scheduling frameworks which avoid 

72 using the global random state. See e.g. #1709. 

73 """ 

74 assert isinstance(seed, integer_types) and 0 <= seed < 2 ** 32 

75 states = [] # type: list 

76 

77 def seed_all(): 

78 assert not states 

79 for r in RANDOMS_TO_MANAGE: 

80 states.append(r.getstate()) 

81 r.seed(seed) 

82 

83 def restore_all(): 

84 assert len(states) == len(RANDOMS_TO_MANAGE) 

85 for r, state in zip(RANDOMS_TO_MANAGE, states): 

86 r.setstate(state) 

87 del states[:] 

88 

89 return seed_all, restore_all

92@contextlib.contextmanager 

93def deterministic_PRNG(): 

94 """Context manager that handles random.seed without polluting global state. 

95 

96 See issue #1255 and PR #1295 for details and motivation - in short, 

97 leaving the global pseudo-random number generator (PRNG) seeded is a very 

98 bad idea in principle, and breaks all kinds of independence assumptions 

99 in practice. 

100 """ 

101 seed_all, restore_all = get_seeder_and_restorer() 

102 seed_all() 

103 try: 

104 yield 

105 finally: 

106 restore_all()
