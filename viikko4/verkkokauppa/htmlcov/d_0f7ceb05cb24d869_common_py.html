25def find_integer(f): 

26 """Finds a (hopefully large) integer such that f(n) is True and f(n + 1) is 

27 False. 

28 

29 f(0) is assumed to be True and will not be checked. 

30 """ 

31 # We first do a linear scan over the small numbers and only start to do 

32 # anything intelligent if f(4) is true. This is because it's very hard to 

33 # win big when the result is small. If the result is 0 and we try 2 first 

34 # then we've done twice as much work as we needed to! 

35 for i in range(1, 5): 

36 if not f(i): 

37 return i - 1 

38 

39 # We now know that f(4) is true. We want to find some number for which 

40 # f(n) is *not* true. 

41 # lo is the largest number for which we know that f(lo) is true. 

42 lo = 4 

43 

44 # Exponential probe upwards until we find some value hi such that f(hi) 

45 # is not true. Subsequently we maintain the invariant that hi is the 

46 # smallest number for which we know that f(hi) is not true. 

47 hi = 5 

48 while f(hi): 

49 lo = hi 

50 hi *= 2 

51 

52 # Now binary search until lo + 1 = hi. At that point we have f(lo) and not 

53 # f(lo + 1), as desired.. 

54 while lo + 1 < hi: 

55 mid = (lo + hi) // 2 

56 if f(mid): 

57 lo = mid 

58 else: 

59 hi = mid 

60 return lo

136 def run(self): 

137 """Run for an appropriate number of steps to improve the current value. 

138 

139 If self.full is True, will run until no further improvements can 

140 be found. 

141 """ 

142 if self.short_circuit(): 

143 return 

144 if self.full: 

145 prev = -1 

146 while self.changes != prev: 

147 prev = self.changes 

148 self.run_step() 

149 else: 

150 self.run_step() 

151 self.debug("COMPLETE")

153 def incorporate(self, value): 

154 """Try using ``value`` as a possible candidate improvement. 

155 

156 Return True if it works. 

157 """ 

158 value = self.make_immutable(value) 

159 self.check_invariants(value) 

160 if not self.left_is_better(value, self.current): 

161 if value != self.current and (value == value): 

162 self.debug( 

163 "Rejected %r as worse than self.current=%r" % (value, self.current) 

164 ) 

165 return False 

166 if value in self.__seen: 

167 return False 

168 self.__seen.add(value) 

169 if self.__predicate(value): 

170 self.debug("shrinking to %r" % (value,)) 

171 self.changes += 1 

172 self.current = value 

173 return True 

174 return False

176 def consider(self, value): 

177 """Returns True if make_immutable(value) == self.current after calling 

178 self.incorporate(value).""" 

179 value = self.make_immutable(value) 

180 if value == self.current: 

181 return True 

182 return self.incorporate(value)
