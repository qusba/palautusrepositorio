78def format_exception(msg, *args, **kwargs): 

79 """ 

80 Given an exception message string, uses new-style formatting arguments 

81 ``{filename}``, ``{lineno}``, ``{func}`` and/or ``{text}`` to fill in 

82 information about the exception that occurred. For example: 

83 

84 try: 

85 1/0 

86 except: 

87 raise ZeroDivisionError( 

88 format_except('A divide by zero occurred in {filename} at ' 

89 'line {lineno} of function {func}.')) 

90 

91 Any additional positional or keyword arguments passed to this function are 

92 also used to format the message. 

93 

94 .. note:: 

95 This uses `sys.exc_info` to gather up the information needed to fill 

96 in the formatting arguments. Since `sys.exc_info` is not carried 

97 outside a handled exception, it's not wise to use this 

98 outside of an ``except`` clause - if it is, this will substitute 

99 '<unkonwn>' for the 4 formatting arguments. 

100 """ 

101 

102 tb = traceback.extract_tb(sys.exc_info()[2], limit=1) 

103 if len(tb) > 0: 

104 filename, lineno, func, text = tb[0] 

105 else: 

106 filename = lineno = func = text = '<unknown>' 

107 

108 return msg.format(*args, filename=filename, lineno=lineno, func=func, 

109 text=text, **kwargs)

159def find_api_page(obj, version=None, openinbrowser=True, timeout=None): 

160 """ 

161 Determines the URL of the API page for the specified object, and 

162 optionally open that page in a web browser. 

163 

164 .. note:: 

165 You must be connected to the internet for this to function even if 

166 ``openinbrowser`` is `False`, unless you provide a local version of 

167 the documentation to ``version`` (e.g., ``file:///path/to/docs``). 

168 

169 Parameters 

170 ---------- 

171 obj 

172 The object to open the docs for or its fully-qualified name 

173 (as a str). 

174 version : str 

175 The doc version - either a version number like '0.1', 'dev' for 

176 the development/latest docs, or a URL to point to a specific 

177 location that should be the *base* of the documentation. Defaults to 

178 latest if you are on aren't on a release, otherwise, the version you 

179 are on. 

180 openinbrowser : bool 

181 If `True`, the `webbrowser` package will be used to open the doc 

182 page in a new web browser window. 

183 timeout : number, optional 

184 The number of seconds to wait before timing-out the query to 

185 the astropy documentation. If not given, the default python 

186 stdlib timeout will be used. 

187 

188 Returns 

189 ------- 

190 url : str 

191 The loaded URL 

192 

193 Raises 

194 ------ 

195 ValueError 

196 If the documentation can't be found 

197 

198 """ 

199 import webbrowser 

200 import urllib.request 

201 from zlib import decompress 

202 

203 if (not isinstance(obj, str) and 

204 hasattr(obj, '__module__') and 

205 hasattr(obj, '__name__')): 

206 obj = obj.__module__ + '.' + obj.__name__ 

207 elif inspect.ismodule(obj): 

208 obj = obj.__name__ 

209 

210 if version is None: 

211 from astropy import version 

212 

213 if version.release: 

214 version = 'v' + version.version 

215 else: 

216 version = 'dev' 

217 

218 if '://' in version: 

219 if version.endswith('index.html'): 

220 baseurl = version[:-10] 

221 elif version.endswith('/'): 

222 baseurl = version 

223 else: 

224 baseurl = version + '/' 

225 elif version == 'dev' or version == 'latest': 

226 baseurl = 'http://devdocs.astropy.org/' 

227 else: 

228 baseurl = f'https://docs.astropy.org/en/{version}/' 

229 

230 # Custom request headers; see 

231 # https://github.com/astropy/astropy/issues/8990 

232 req = urllib.request.Request( 

233 baseurl + 'objects.inv', headers={'User-Agent': f'Astropy/{version}'}) 

234 

235 if timeout is None: 

236 uf = urllib.request.urlopen(req) 

237 else: 

238 uf = urllib.request.urlopen(req, timeout=timeout) 

239 

240 try: 

241 oiread = uf.read() 

242 

243 # need to first read/remove the first four lines, which have info before 

244 # the compressed section with the actual object inventory 

245 idx = -1 

246 headerlines = [] 

247 for _ in range(4): 

248 oldidx = idx 

249 idx = oiread.index(b'\n', oldidx + 1) 

250 headerlines.append(oiread[(oldidx+1):idx].decode('utf-8')) 

251 

252 # intersphinx version line, project name, and project version 

253 ivers, proj, vers, compr = headerlines 

254 if 'The remainder of this file is compressed using zlib' not in compr: 

255 raise ValueError('The file downloaded from {} does not seem to be' 

256 'the usual Sphinx objects.inv format. Maybe it ' 

257 'has changed?'.format(baseurl + 'objects.inv')) 

258 

259 compressed = oiread[(idx+1):] 

260 finally: 

261 uf.close() 

262 

263 decompressed = decompress(compressed).decode('utf-8') 

264 

265 resurl = None 

266 

267 for l in decompressed.strip().splitlines(): 

268 ls = l.split() 

269 name = ls[0] 

270 loc = ls[3] 

271 if loc.endswith('$'): 

272 loc = loc[:-1] + name 

273 

274 if name == obj: 

275 resurl = baseurl + loc 

276 break 

277 

278 if resurl is None: 

279 raise ValueError(f'Could not find the docs for the object {obj}') 

280 elif openinbrowser: 

281 webbrowser.open(resurl) 

282 

283 return resurl

422def did_you_mean(s, candidates, n=3, cutoff=0.8, fix=None): 

423 """ 

424 When a string isn't found in a set of candidates, we can be nice 

425 to provide a list of alternatives in the exception. This 

426 convenience function helps to format that part of the exception. 

427 

428 Parameters 

429 ---------- 

430 s : str 

431 

432 candidates : sequence of str or dict of str keys 

433 

434 n : int 

435 The maximum number of results to include. See 

436 `difflib.get_close_matches`. 

437 

438 cutoff : float 

439 In the range [0, 1]. Possibilities that don't score at least 

440 that similar to word are ignored. See 

441 `difflib.get_close_matches`. 

442 

443 fix : callable 

444 A callable to modify the results after matching. It should 

445 take a single string and return a sequence of strings 

446 containing the fixed matches. 

447 

448 Returns 

449 ------- 

450 message : str 

451 Returns the string "Did you mean X, Y, or Z?", or the empty 

452 string if no alternatives were found. 

453 """ 

454 if isinstance(s, str): 

455 s = strip_accents(s) 

456 s_lower = s.lower() 

457 

458 # Create a mapping from the lower case name to all capitalization 

459 # variants of that name. 

460 candidates_lower = {} 

461 for candidate in candidates: 

462 candidate_lower = candidate.lower() 

463 candidates_lower.setdefault(candidate_lower, []) 

464 candidates_lower[candidate_lower].append(candidate) 

465 

466 # The heuristic here is to first try "singularizing" the word. If 

467 # that doesn't match anything use difflib to find close matches in 

468 # original, lower and upper case. 

469 if s_lower.endswith('s') and s_lower[:-1] in candidates_lower: 

470 matches = [s_lower[:-1]] 

471 else: 

472 matches = difflib.get_close_matches( 

473 s_lower, candidates_lower, n=n, cutoff=cutoff) 

474 

475 if len(matches): 

476 capitalized_matches = set() 

477 for match in matches: 

478 capitalized_matches.update(candidates_lower[match]) 

479 matches = capitalized_matches 

480 

481 if fix is not None: 

482 mapped_matches = [] 

483 for match in matches: 

484 mapped_matches.extend(fix(match)) 

485 matches = mapped_matches 

486 

487 matches = list(set(matches)) 

488 matches = sorted(matches) 

489 

490 if len(matches) == 1: 

491 matches = matches[0] 

492 else: 

493 matches = (', '.join(matches[:-1]) + ' or ' + 

494 matches[-1]) 

495 return f'Did you mean {matches}?' 

496 

497 return ''

533 return (

536 not key.startswith('_')) 

537 

538 for key, val in dct.items():

551class OrderedDescriptor(metaclass=abc.ABCMeta): 

552 """ 

553 Base class for descriptors whose order in the class body should be 

554 preserved. Intended for use in concert with the 

555 `OrderedDescriptorContainer` metaclass. 

556 

557 Subclasses of `OrderedDescriptor` must define a value for a class attribute 

558 called ``_class_attribute_``. This is the name of a class attribute on the 

559 *container* class for these descriptors, which will be set to an 

560 `~collections.OrderedDict` at class creation time. This 

561 `~collections.OrderedDict` will contain a mapping of all class attributes 

562 that were assigned instances of the `OrderedDescriptor` subclass, to the 

563 instances themselves. See the documentation for 

564 `OrderedDescriptorContainer` for a concrete example. 

565 

566 Optionally, subclasses of `OrderedDescriptor` may define a value for a 

567 class attribute called ``_name_attribute_``. This should be the name of 

568 an attribute on instances of the subclass. When specified, during 

569 creation of a class containing these descriptors, the name attribute on 

570 each instance will be set to the name of the class attribute it was 

571 assigned to on the class. 

572 

573 .. note:: 

574 

575 Although this class is intended for use with *descriptors* (i.e. 

576 classes that define any of the ``__get__``, ``__set__``, or 

577 ``__delete__`` magic methods), this base class is not itself a 

578 descriptor, and technically this could be used for classes that are 

579 not descriptors too. However, use with descriptors is the original 

580 intended purpose. 

581 """ 

582 

583 # This id increments for each OrderedDescriptor instance created, so they 

584 # are always ordered in the order they were created. Class bodies are 

585 # guaranteed to be executed from top to bottom. Not sure if this is 

586 # thread-safe though. 

587 _nextid = 1 

588 

589 @property 

590 @abc.abstractmethod 

591 def _class_attribute_(self): 

592 """ 

593 Subclasses should define this attribute to the name of an attribute on 

594 classes containing this subclass. That attribute will contain the mapping 

595 of all instances of that `OrderedDescriptor` subclass defined in the class 

596 body. If the same descriptor needs to be used with different classes, 

597 each with different names of this attribute, multiple subclasses will be 

598 needed. 

599 """ 

600 

601 _name_attribute_ = None 

602 """ 

603 Subclasses may optionally define this attribute to specify the name of an 

604 attribute on instances of the class that should be filled with the 

605 instance's attribute name at class creation time. 

606 """ 

607 

608 def __init__(self, *args, **kwargs): 

609 # The _nextid attribute is shared across all subclasses so that 

610 # different subclasses of OrderedDescriptors can be sorted correctly 

611 # between themselves 

612 self.__order = OrderedDescriptor._nextid 

613 OrderedDescriptor._nextid += 1 

614 super().__init__() 

615 

616 def __lt__(self, other): 

617 """ 

618 Defined for convenient sorting of `OrderedDescriptor` instances, which 

619 are defined to sort in their creation order. 

620 """ 

621 

622 if (isinstance(self, OrderedDescriptor) and 

623 isinstance(other, OrderedDescriptor)): 

624 try: 

625 return self.__order < other.__order 

626 except AttributeError: 

627 raise RuntimeError( 

628 'Could not determine ordering for {} and {}; at least ' 

629 'one of them is not calling super().__init__ in its ' 

630 '__init__.'.format(self, other)) 

631 else: 

632 return NotImplemented

635class OrderedDescriptorContainer(type): 

636 """ 

637 Classes should use this metaclass if they wish to use `OrderedDescriptor` 

638 attributes, which are class attributes that "remember" the order in which 

639 they were defined in the class body. 

640 

641 Every subclass of `OrderedDescriptor` has an attribute called 

642 ``_class_attribute_``. For example, if we have 

643 

644 .. code:: python 

645 

646 class ExampleDecorator(OrderedDescriptor): 

647 _class_attribute_ = '_examples_' 

648 

649 Then when a class with the `OrderedDescriptorContainer` metaclass is 

650 created, it will automatically be assigned a class attribute ``_examples_`` 

651 referencing an `~collections.OrderedDict` containing all instances of 

652 ``ExampleDecorator`` defined in the class body, mapped to by the names of 

653 the attributes they were assigned to. 

654 

655 When subclassing a class with this metaclass, the descriptor dict (i.e. 

656 ``_examples_`` in the above example) will *not* contain descriptors 

657 inherited from the base class. That is, this only works by default with 

658 decorators explicitly defined in the class body. However, the subclass 

659 *may* define an attribute ``_inherit_decorators_`` which lists 

660 `OrderedDescriptor` classes that *should* be added from base classes. 

661 See the examples section below for an example of this. 

662 

663 Examples 

664 -------- 

665 

666 >>> from astropy.utils import OrderedDescriptor, OrderedDescriptorContainer 

667 >>> class TypedAttribute(OrderedDescriptor): 

668 ... \"\"\" 

669 ... Attributes that may only be assigned objects of a specific type, 

670 ... or subclasses thereof. For some reason we care about their order. 

671 ... \"\"\" 

672 ... 

673 ... _class_attribute_ = 'typed_attributes' 

674 ... _name_attribute_ = 'name' 

675 ... # A default name so that instances not attached to a class can 

676 ... # still be repr'd; useful for debugging 

677 ... name = '<unbound>' 

678 ... 

679 ... def __init__(self, type): 

680 ... # Make sure not to forget to call the super __init__ 

681 ... super().__init__() 

682 ... self.type = type 

683 ... 

684 ... def __get__(self, obj, objtype=None): 

685 ... if obj is None: 

686 ... return self 

687 ... if self.name in obj.__dict__: 

688 ... return obj.__dict__[self.name] 

689 ... else: 

690 ... raise AttributeError(self.name) 

691 ... 

692 ... def __set__(self, obj, value): 

693 ... if not isinstance(value, self.type): 

694 ... raise ValueError('{0}.{1} must be of type {2!r}'.format( 

695 ... obj.__class__.__name__, self.name, self.type)) 

696 ... obj.__dict__[self.name] = value 

697 ... 

698 ... def __delete__(self, obj): 

699 ... if self.name in obj.__dict__: 

700 ... del obj.__dict__[self.name] 

701 ... else: 

702 ... raise AttributeError(self.name) 

703 ... 

704 ... def __repr__(self): 

705 ... if isinstance(self.type, tuple) and len(self.type) > 1: 

706 ... typestr = '({0})'.format( 

707 ... ', '.join(t.__name__ for t in self.type)) 

708 ... else: 

709 ... typestr = self.type.__name__ 

710 ... return '<{0}(name={1}, type={2})>'.format( 

711 ... self.__class__.__name__, self.name, typestr) 

712 ... 

713 

714 Now let's create an example class that uses this ``TypedAttribute``:: 

715 

716 >>> class Point2D(metaclass=OrderedDescriptorContainer): 

717 ... x = TypedAttribute((float, int)) 

718 ... y = TypedAttribute((float, int)) 

719 ... 

720 ... def __init__(self, x, y): 

721 ... self.x, self.y = x, y 

722 ... 

723 >>> p1 = Point2D(1.0, 2.0) 

724 >>> p1.x 

725 1.0 

726 >>> p1.y 

727 2.0 

728 >>> p2 = Point2D('a', 'b') # doctest: +IGNORE_EXCEPTION_DETAIL 

729 Traceback (most recent call last): 

730 ... 

731 ValueError: Point2D.x must be of type (float, int>) 

732 

733 We see that ``TypedAttribute`` works more or less as advertised, but 

734 there's nothing special about that. Let's see what 

735 `OrderedDescriptorContainer` did for us:: 

736 

737 >>> Point2D.typed_attributes 

738 OrderedDict([('x', <TypedAttribute(name=x, type=(float, int))>), 

739 ('y', <TypedAttribute(name=y, type=(float, int))>)]) 

740 

741 If we create a subclass, it does *not* by default add inherited descriptors 

742 to ``typed_attributes``:: 

743 

744 >>> class Point3D(Point2D): 

745 ... z = TypedAttribute((float, int)) 

746 ... 

747 >>> Point3D.typed_attributes 

748 OrderedDict([('z', <TypedAttribute(name=z, type=(float, int))>)]) 

749 

750 However, if we specify ``_inherit_descriptors_`` from ``Point2D`` then 

751 it will do so:: 

752 

753 >>> class Point3D(Point2D): 

754 ... _inherit_descriptors_ = (TypedAttribute,) 

755 ... z = TypedAttribute((float, int)) 

756 ... 

757 >>> Point3D.typed_attributes 

758 OrderedDict([('x', <TypedAttribute(name=x, type=(float, int))>), 

759 ('y', <TypedAttribute(name=y, type=(float, int))>), 

760 ('z', <TypedAttribute(name=z, type=(float, int))>)]) 

761 

762 .. note:: 

763 

764 Hopefully it is clear from these examples that this construction 

765 also allows a class of type `OrderedDescriptorContainer` to use 

766 multiple different `OrderedDescriptor` classes simultaneously. 

767 """ 

768 

769 _inherit_descriptors_ = () 

770 

771 def __init__(cls, cls_name, bases, members): 

772 descriptors = defaultdict(list) 

773 seen = set() 

774 inherit_descriptors = () 

775 descr_bases = {} 

776 

777 for mro_cls in cls.__mro__: 

778 for name, obj in mro_cls.__dict__.items(): 

779 if name in seen: 

780 # Checks if we've already seen an attribute of the given 

781 # name (if so it will override anything of the same name in 

782 # any base class) 

783 continue 

784 

785 seen.add(name) 

786 

787 if (not isinstance(obj, OrderedDescriptor) or 

788 (inherit_descriptors and 

789 not isinstance(obj, inherit_descriptors))): 

790 # The second condition applies when checking any 

791 # subclasses, to see if we can inherit any descriptors of 

792 # the given type from subclasses (by default inheritance is 

793 # disabled unless the class has _inherit_descriptors_ 

794 # defined) 

795 continue 

796 

797 if obj._name_attribute_ is not None: 

798 setattr(obj, obj._name_attribute_, name) 

799 

800 # Don't just use the descriptor's class directly; instead go 

801 # through its MRO and find the class on which _class_attribute_ 

802 # is defined directly. This way subclasses of some 

803 # OrderedDescriptor *may* override _class_attribute_ and have 

804 # its own _class_attribute_, but by default all subclasses of 

805 # some OrderedDescriptor are still grouped together 

806 # TODO: It might be worth clarifying this in the docs 

807 if obj.__class__ not in descr_bases: 

808 for obj_cls_base in obj.__class__.__mro__: 

809 if '_class_attribute_' in obj_cls_base.__dict__: 

810 descr_bases[obj.__class__] = obj_cls_base 

811 descriptors[obj_cls_base].append((obj, name)) 

812 break 

813 else: 

814 # Make sure to put obj first for sorting purposes 

815 obj_cls_base = descr_bases[obj.__class__] 

816 descriptors[obj_cls_base].append((obj, name)) 

817 

818 if not getattr(mro_cls, '_inherit_descriptors_', False): 

819 # If _inherit_descriptors_ is undefined then we don't inherit 

820 # any OrderedDescriptors from any of the base classes, and 

821 # there's no reason to continue through the MRO 

822 break 

823 else: 

824 inherit_descriptors = mro_cls._inherit_descriptors_ 

825 

826 for descriptor_cls, instances in descriptors.items(): 

827 instances.sort() 

828 instances = OrderedDict((key, value) for value, key in instances) 

829 setattr(cls, descriptor_cls._class_attribute_, instances) 

830 

831 super(OrderedDescriptorContainer, cls).__init__(cls_name, bases, 

832 members)

898class ShapedLikeNDArray(metaclass=abc.ABCMeta): 

899 """Mixin class to provide shape-changing methods. 

900 

901 The class proper is assumed to have some underlying data, which are arrays 

902 or array-like structures. It must define a ``shape`` property, which gives 

903 the shape of those data, as well as an ``_apply`` method that creates a new 

904 instance in which a `~numpy.ndarray` method has been applied to those. 

905 

906 Furthermore, for consistency with `~numpy.ndarray`, it is recommended to 

907 define a setter for the ``shape`` property, which, like the 

908 `~numpy.ndarray.shape` property allows in-place reshaping the internal data 

909 (and, unlike the ``reshape`` method raises an exception if this is not 

910 possible). 

911 

912 This class also defines default implementations for ``ndim`` and ``size`` 

913 properties, calculating those from the ``shape``. These can be overridden 

914 by subclasses if there are faster ways to obtain those numbers. 

915 

916 """ 

917 

918 # Note to developers: if new methods are added here, be sure to check that 

919 # they work properly with the classes that use this, such as Time and 

920 # BaseRepresentation, i.e., look at their ``_apply`` methods and add 

921 # relevant tests. This is particularly important for methods that imply 

922 # copies rather than views of data (see the special-case treatment of 

923 # 'flatten' in Time). 

924 

925 @property 

926 @abc.abstractmethod 

927 def shape(self): 

928 """The shape of the underlying data.""" 

929 

930 @abc.abstractmethod 

931 def _apply(method, *args, **kwargs): 

932 """Create a new instance, with ``method`` applied to underlying data. 

933 

934 The method is any of the shape-changing methods for `~numpy.ndarray` 

935 (``reshape``, ``swapaxes``, etc.), as well as those picking particular 

936 elements (``__getitem__``, ``take``, etc.). It will be applied to the 

937 underlying arrays (e.g., ``jd1`` and ``jd2`` in `~astropy.time.Time`), 

938 with the results used to create a new instance. 

939 

940 Parameters 

941 ---------- 

942 method : str 

943 Method to be applied to the instance's internal data arrays. 

944 args : tuple 

945 Any positional arguments for ``method``. 

946 kwargs : dict 

947 Any keyword arguments for ``method``. 

948 

949 """ 

950 

951 @property 

952 def ndim(self): 

953 """The number of dimensions of the instance and underlying arrays.""" 

954 return len(self.shape) 

955 

956 @property 

957 def size(self): 

958 """The size of the object, as calculated from its shape.""" 

959 size = 1 

960 for sh in self.shape: 

961 size *= sh 

962 return size

964 @property

966 return self.shape == ()

969 if self.isscalar: 

970 raise TypeError("Scalar {!r} object has no len()" 

971 .format(self.__class__.__name__)) 

972 return self.shape[0] 

973 

974 def __bool__(self): 

975 """Any instance should evaluate to True, except when it is empty.""" 

976 return self.size > 0 

977 

978 def __getitem__(self, item): 

979 try:

981 except IndexError: 

982 if self.isscalar:

1097def check_broadcast(*shapes): 

1098 """ 

1099 Determines whether two or more Numpy arrays can be broadcast with each 

1100 other based on their shape tuple alone. 

1101 

1102 Parameters 

1103 ---------- 

1104 *shapes : tuple 

1105 All shapes to include in the comparison. If only one shape is given it 

1106 is passed through unmodified. If no shapes are given returns an empty 

1107 `tuple`. 

1108 

1109 Returns 

1110 ------- 

1111 broadcast : `tuple` 

1112 If all shapes are mutually broadcastable, returns a tuple of the full 

1113 broadcast shape. 

1114 """ 

1115 

1116 if len(shapes) == 0: 

1117 return () 

1118 elif len(shapes) == 1: 

1119 return shapes[0] 

1120 

1121 reversed_shapes = (reversed(shape) for shape in shapes) 

1122 

1123 full_shape = [] 

1124 

1125 for dims in zip_longest(*reversed_shapes, fillvalue=1): 

1126 max_dim = 1 

1127 max_dim_idx = None 

1128 for idx, dim in enumerate(dims): 

1129 if dim == 1: 

1130 continue 

1131 

1132 if max_dim == 1: 

1133 # The first dimension of size greater than 1 

1134 max_dim = dim 

1135 max_dim_idx = idx 

1136 elif dim != max_dim: 

1137 raise IncompatibleShapeError( 

1138 shapes[max_dim_idx], max_dim_idx, shapes[idx], idx) 

1139 

1140 full_shape.append(max_dim) 

1141 

1142 return tuple(full_shape[::-1])

1167def unbroadcast(array): 

1168 """ 

1169 Given an array, return a new array that is the smallest subset of the 

1170 original array that can be re-broadcasted back to the original array. 

1171 

1172 See https://stackoverflow.com/questions/40845769/un-broadcasting-numpy-arrays 

1173 for more details. 

1174 """ 

1175 

1176 if array.ndim == 0: 

1177 return array 

1178 

1179 array = array[tuple((slice(0, 1) if stride == 0 else slice(None)) 

1180 for stride in array.strides)] 

1181 

1182 # Remove leading ones, which are not needed in numpy broadcasting. 

1183 first_not_unity = next((i for (i, s) in enumerate(array.shape) if s > 1), 

1184 array.ndim) 

1185 

1186 return array.reshape(array.shape[first_not_unity:])
