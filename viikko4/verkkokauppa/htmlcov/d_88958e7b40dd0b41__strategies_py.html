1490 def fraction_bounds(value): 

1491 # type: (Fraction) -> Tuple[Fraction, Fraction] 

1492 """Find the best lower and upper approximation for value.""" 

1493 # Adapted from CPython's Fraction.limit_denominator here: 

1494 # https://github.com/python/cpython/blob/3.6/Lib/fractions.py#L219 

1495 assert max_denominator is not None 

1496 if value is None or value.denominator <= max_denominator: 

1497 return value, value 

1498 

1499 p0, q0, p1, q1 = 0, 1, 1, 0 

1500 n, d = value.numerator, value.denominator 

1501 while True: 

1502 a = n // d 

1503 q2 = q0 + a * q1 

1504 if q2 > max_denominator: 

1505 break 

1506 p0, q0, p1, q1 = p1, q1, p0 + a * p1, q2 

1507 n, d = d, n - a * d 

1508 k = (max_denominator - q0) // q1 

1509 low, high = Fraction(p1, q1), Fraction(p0 + k * p1, q0 + k * q1) 

1510 assert low < value < high 

1511 return low, high 

1512 

1513 # Take the high approximation for min_value and low for max_value 

1514 bounds = (max_denominator, min_value, max_value) 

1515 if min_value is not None: 

1516 if min_value.denominator > max_denominator: 

1517 note_deprecation( 

1518 "The min_value=%r has a denominator greater than the " 

1519 "max_denominator=%r, which will be an error in a future " 

1520 "version." % (min_value, max_denominator), 

1521 since="2018-10-12", 

1522 ) 

1523 _, min_value = fraction_bounds(min_value) 

1524 if max_value is not None: 

1525 if max_value.denominator > max_denominator: 

1526 note_deprecation( 

1527 "The max_value=%r has a denominator greater than the " 

1528 "max_denominator=%r, which will be an error in a future " 

1529 "version." % (max_value, max_denominator), 

1530 since="2018-10-12", 

1531 ) 

1532 max_value, _ = fraction_bounds(max_value) 

1533 

1534 if min_value is not None and max_value is not None and min_value > max_value: 

1535 raise InvalidArgument( 

1536 "There are no fractions with a denominator <= %r between " 

1537 "min_value=%r and max_value=%r" % bounds 

1538 ) 

1539 

1540 if min_value is not None and min_value == max_value: 

1541 return just(min_value)

1543 def dm_func(denom): 

1544 """Take denom, construct numerator strategy, and build fraction.""" 

1545 # Four cases of algebra to get integer bounds and scale factor. 

1546 min_num, max_num = None, None 

1547 if max_value is None and min_value is None: 

1548 pass 

1549 elif min_value is None: 

1550 max_num = denom * max_value.numerator 

1551 denom *= max_value.denominator 

1552 elif max_value is None: 

1553 min_num = denom * min_value.numerator 

1554 denom *= min_value.denominator 

1555 else: 

1556 low = min_value.numerator * max_value.denominator 

1557 high = max_value.numerator * min_value.denominator 

1558 scale = min_value.denominator * max_value.denominator 

1559 # After calculating our integer bounds and scale factor, we remove 

1560 # the gcd to avoid drawing more bytes for the example than needed. 

1561 # Note that `div` can be at most equal to `scale`. 

1562 div = gcd(scale, gcd(low, high)) 

1563 min_num = denom * low // div 

1564 max_num = denom * high // div 

1565 denom *= scale // div 

1566 

1567 return builds( 

1568 Fraction, integers(min_value=min_num, max_value=max_num), just(denom) 

1569 ) 

1570 

1571 if max_denominator is None: 

1572 return integers(min_value=1).flatmap(dm_func) 

1573 

1574 return ( 

1575 integers(1, max_denominator) 

1576 .flatmap(dm_func) 

1577 .map(lambda f: f.limit_denominator(max_denominator)) 

1578 )
