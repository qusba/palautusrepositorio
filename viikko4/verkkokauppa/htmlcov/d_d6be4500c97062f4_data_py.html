67class Example(object): 

68 """Examples track the hierarchical structure of draws from the byte stream, 

69 within a single test run. 

70 

71 Examples are created to mark regions of the byte stream that might be 

72 useful to the shrinker, such as: 

73 - The bytes used by a single draw from a strategy. 

74 - Useful groupings within a strategy, such as individual list elements. 

75 - Strategy-like helper functions that aren't first-class strategies. 

76 - Each lowest-level draw of bits or bytes from the byte stream. 

77 - A single top-level example that spans the entire input. 

78 

79 Example-tracking allows the shrinker to try "high-level" transformations, 

80 such as rearranging or deleting the elements of a list, without having 

81 to understand their exact representation in the byte stream. 

82 

83 Rather than store each ``Example`` as a rich object, it is actually 

84 just an index into the ``Examples`` class defined below. This has two 

85 purposes: Firstly, for most properties of examples we will never need 

86 to allocate storage at all, because most properties are not used on 

87 most examples. Secondly, by storing the properties as compact lists 

88 of integers, we save a considerable amount of space compared to 

89 Python's normal object size. 

90 

91 This does have the downside that it increases the amount of allocation 

92 we do, and slows things down as a result, in some usage patterns because 

93 we repeatedly allocate the same Example or int objects, but it will 

94 often dramatically reduce our memory usage, so is worth it. 

95 """ 

96 

97 __slots__ = ("owner", "index") 

98

100 self.owner = owner 

101 self.index = index 

102 

103 def __eq__(self, other):

105 return True 

106 if not isinstance(other, Example): 

107 return NotImplemented 

108 return (self.owner is other.owner) and (self.index == other.index) 

109 

110 def __ne__(self, other): 

111 if self is other: 

112 return False 

113 if not isinstance(other, Example): 

114 return NotImplemented 

115 return (self.owner is not other.owner) or (self.index != other.index)

142 i.e. the example corresponds to the half open region [start, end).

157 return self.index in self.owner.trivial

180class ExampleProperty(object): 

181 """There are many properties of examples that we calculate by 

182 essentially rerunning thet test case multiple times based on the 

183 calls which we record in ExampleRecord. 

184 

185 This class defines a visitor, subclasses of which can be used 

186 to calculate these properties. 

187 """ 

188 

189 def __init__(self, examples): 

190 self.example_stack = [] 

191 self.examples = examples 

192 self.bytes_read = 0 

193 self.example_count = 0 

194 self.block_count = 0 

195 

196 def run(self): 

197 """Rerun the test case with this visitor and return the 

198 results of ``self.finish()``.""" 

199 self.begin() 

200 blocks = self.examples.blocks 

201 for record in self.examples.trail: 

202 if record == DRAW_BITS_RECORD: 

203 self.__push(0) 

204 self.bytes_read = blocks.endpoints[self.block_count] 

205 self.block(self.block_count) 

206 self.block_count += 1 

207 self.__pop(False) 

208 elif record >= START_EXAMPLE_RECORD: 

209 self.__push(record - START_EXAMPLE_RECORD) 

210 else: 

211 assert record in ( 

212 STOP_EXAMPLE_DISCARD_RECORD, 

213 STOP_EXAMPLE_NO_DISCARD_RECORD, 

214 ) 

215 self.__pop(record == STOP_EXAMPLE_DISCARD_RECORD) 

216 return self.finish() 

217 

218 def __push(self, label_index): 

219 i = self.example_count 

220 assert i < len(self.examples) 

221 self.start_example(i, label_index) 

222 self.example_count += 1 

223 self.example_stack.append(i) 

224 

225 def __pop(self, discarded): 

226 i = self.example_stack.pop() 

227 self.stop_example(i, discarded) 

228 

229 def begin(self): 

230 """Called at the beginning of the run to initialise any 

231 relevant state.""" 

232 self.result = IntList.of_length(len(self.examples)) 

233 

234 def start_example(self, i, label_index): 

235 """Called at the start of each example, with ``i`` the 

236 index of the example and ``label_index`` the index of 

237 its label in ``self.examples.labels``.""" 

238 

239 def block(self, i): 

240 """Called with each ``draw_bits`` call, with ``i`` the index of the 

241 corresonding block in ``self.examples.blocks``""" 

242 

243 def stop_example(self, i, discarded): 

244 """Called at the end of each example, with ``i`` the 

245 index of the example and ``discarded`` being ``True`` if ``stop_example`` 

246 was called with ``discard=True``.""" 

247 

248 def finish(self): 

249 return self.result

252def calculated_example_property(cls): 

253 """Given an ``ExampleProperty`` as above we use this decorator 

254 to transform it into a lazy property on the ``Examples`` class, 

255 which has as its value the result of calling ``cls.run()``, 

256 computed the first time the property is accessed. 

257 

258 This has the slightly weird result that we are defining nested 

259 classes which get turned into properties.""" 

260 name = cls.__name__ 

261 cache_name = "__" + name 

262 

263 def lazy_calculate(self): 

264 result = getattr(self, cache_name, None) 

265 if result is None: 

266 result = cls(self).run() 

267 setattr(self, cache_name, result) 

268 return result 

269 

270 lazy_calculate.__name__ = cls.__name__ 

271 lazy_calculate.__qualname__ = getattr(cls, "__qualname__", cls.__name__) 

272 return property(lazy_calculate)

281class ExampleRecord(object): 

282 """Records the series of ``start_example``, ``stop_example``, and 

283 ``draw_bits`` calls so that these may be stored in ``Examples`` and 

284 replayed when we need to know about the structure of individual 

285 ``Example`` objects. 

286 

287 Note that there is significant similarity between this class and 

288 ``DataObserver``, and the plan is to eventually unify them, but 

289 they currently have slightly different functions and implementations. 

290 """ 

291 

292 def __init__(self): 

293 self.labels = [DRAW_BYTES_LABEL] 

294 self.__index_of_labels = {DRAW_BYTES_LABEL: 0} 

295 self.trail = IntList() 

296 

297 def freeze(self): 

298 self.__index_of_labels = None 

299 

300 def start_example(self, label): 

301 try: 

302 i = self.__index_of_labels[label] 

303 except KeyError: 

304 i = self.__index_of_labels.setdefault(label, len(self.labels)) 

305 self.labels.append(label) 

306 self.trail.append(START_EXAMPLE_RECORD + i) 

307

309 if discard: 

310 self.trail.append(STOP_EXAMPLE_DISCARD_RECORD)

481class Blocks(object): 

482 """A lazily calculated list of blocks for a particular ``ConjectureResult`` 

483 or ``ConjectureData`` object. 

484 

485 Pretends to be a list containing ``Block`` objects but actually only 

486 contains their endpoints right up until the point where you want to 

487 access the actual block, at which point it is constructed. 

488 

489 This is designed to be as space efficient as possible, so will at 

490 various points silently transform its representation into one 

491 that is better suited for the current access pattern. 

492 

493 In addition, it has a number of convenience methods for accessing 

494 properties of the block object at index ``i`` that should generally 

495 be preferred to using the Block objects directly, as it will not 

496 have to allocate the actual object.""" 

497 

498 __slots__ = ("endpoints", "owner", "__blocks", "__count", "__sparse") 

499 

500 def __init__(self, owner): 

501 self.owner = owner 

502 self.endpoints = IntList() 

503 self.__blocks = {} 

504 self.__count = 0 

505 self.__sparse = True 

506 

507 def add_endpoint(self, n): 

508 """Add n to the list of endpoints.""" 

509 assert isinstance(self.owner, ConjectureData) 

510 self.endpoints.append(n) 

511 

512 def transfer_ownership(self, new_owner): 

513 """Used to move ``Blocks`` over to a ``ConjectureResult`` object 

514 when that is read to be used and we no longer want to keep the 

515 whole ``ConjectureData`` around.""" 

516 assert isinstance(new_owner, ConjectureResult) 

517 self.owner = new_owner 

518 self.__check_completion() 

519 

520 def start(self, i): 

521 """Equivalent to self[i].start.""" 

522 i = self._check_index(i) 

523 

524 if i == 0: 

525 return 0 

526 else: 

527 return self.end(i - 1) 

528 

529 def end(self, i): 

530 """Equivalent to self[i].end.""" 

531 return self.endpoints[i] 

532 

533 def bounds(self, i): 

534 """Equivalent to self[i].bounds.""" 

535 return (self.start(i), self.end(i)) 

536 

537 def all_bounds(self): 

538 """Equivalent to [(b.start, b.end) for b in self].""" 

539 prev = 0 

540 for e in self.endpoints: 

541 yield (prev, e) 

542 prev = e 

543 

544 @property 

545 def last_block_length(self): 

546 return self.end(-1) - self.start(-1) 

547 

548 def __len__(self): 

549 return len(self.endpoints) 

550 

551 def __known_block(self, i): 

552 try: 

553 return self.__blocks[i] 

554 except (KeyError, IndexError): 

555 return None 

556 

557 def trivial(self, i): 

558 """Equivalent to self.blocks[i].trivial.""" 

559 if self.owner is not None: 

560 return self.start(i) in self.owner.forced_indices or not any( 

561 self.owner.buffer[self.start(i) : self.end(i)] 

562 ) 

563 else: 

564 return self[i].trivial 

565 

566 def _check_index(self, i): 

567 n = len(self) 

568 if i < -n or i >= n: 

569 raise IndexError("Index %d out of range [-%d, %d)" % (i, n, n)) 

570 if i < 0: 

571 i += n 

572 return i 

573 

574 def __getitem__(self, i): 

575 i = self._check_index(i) 

576 assert i >= 0 

577 result = self.__known_block(i) 

578 if result is not None: 

579 return result 

580 

581 # We store the blocks as a sparse dict mapping indices to the 

582 # actual result, but this isn't the best representation once we 

583 # stop being sparse and want to use most of the blocks. Switch 

584 # over to a list at that point. 

585 if self.__sparse and len(self.__blocks) * 2 >= len(self): 

586 new_blocks = [None] * len(self) 

587 for k, v in self.__blocks.items(): 

588 new_blocks[k] = v 

589 self.__sparse = False 

590 self.__blocks = new_blocks 

591 assert self.__blocks[i] is None 

592 

593 start = self.start(i) 

594 end = self.end(i) 

595 

596 # We keep track of the number of blocks that have actually been 

597 # instantiated so that when every block that could be instantiated 

598 # has been we know that the list is complete and can throw away 

599 # some data that we no longer need. 

600 self.__count += 1 

601 

602 # Integrity check: We can't have allocated more blocks than we have 

603 # positions for blocks. 

604 assert self.__count <= len(self) 

605 result = Block( 

606 start=start, 

607 end=end, 

608 index=i, 

609 forced=start in self.owner.forced_indices, 

610 all_zero=not any(self.owner.buffer[start:end]), 

611 ) 

612 try: 

613 self.__blocks[i] = result 

614 except IndexError: 

615 assert isinstance(self.__blocks, list) 

616 assert len(self.__blocks) < len(self) 

617 self.__blocks.extend([None] * (len(self) - len(self.__blocks))) 

618 self.__blocks[i] = result 

619 

620 self.__check_completion() 

621 

622 return result 

623 

624 def __check_completion(self): 

625 """The list of blocks is complete if we have created every ``Block`` 

626 object that we currently good and know that no more will be created. 

627 

628 If this happens then we don't need to keep the reference to the 

629 owner around, and delete it so that there is no circular reference. 

630 The main benefit of this is that the gc doesn't need to run to collect 

631 this because normal reference counting is enough. 

632 """ 

633 if self.__count == len(self) and isinstance(self.owner, ConjectureResult): 

634 self.owner = None 

635 

636 def __iter__(self): 

637 for i in hrange(len(self)): 

638 yield self[i] 

639 

640 def __repr__(self): 

641 parts = [] 

642 for i in hrange(len(self)): 

643 b = self.__known_block(i) 

644 if b is None: 

645 parts.append("...") 

646 else: 

647 parts.append(repr(b)) 

648 return "Block([%s])" % (", ".join(parts),)

832 def __draw(self, strategy, label): 

833 at_top_level = self.depth == 0 

834 if label is None: 

835 label = strategy.label 

836 self.start_example(label=label) 

837 try: 

838 if not at_top_level: 

839 return strategy.do_draw(self) 

840 else: 

841 try: 

842 strategy.validate() 

843 start_time = benchmark_time() 

844 try: 

845 return strategy.do_draw(self) 

846 finally: 

847 self.draw_times.append(benchmark_time() - start_time) 

848 except BaseException as e: 

849 mark_for_escalation(e) 

850 raise 

851 finally: 

852 self.stop_example()

854 def start_example(self, label): 

855 self.__assert_not_frozen("start_example") 

856 self.depth += 1 

857 # Logically it would make sense for this to just be 

858 # ``self.depth = max(self.depth, self.max_depth)``, which is what it used to 

859 # be until we ran the code under tracemalloc and found a rather significant 

860 # chunk of allocation was happening here. This was presumably due to varargs 

861 # or the like, but we didn't investigate further given that it was easy 

862 # to fix with this check. 

863 if self.depth > self.max_depth: 

864 self.max_depth = self.depth 

865 self.__example_record.start_example(label) 

866 self.consecutive_discard_counts.append(0) 

867 

868 def stop_example(self, discard=False): 

869 if self.frozen: 

870 return 

871 self.consecutive_discard_counts.pop() 

872 if discard: 

873 self.has_discards = True 

874 self.depth -= 1 

875 assert self.depth >= -1 

876 self.__example_record.stop_example(discard) 

877 if self.consecutive_discard_counts: 

878 # We block long sequences of discards. This helps us avoid performance 

879 # problems where there is rejection sampling. In particular tests which 

880 # have a very small actual state space but use rejection sampling will 

881 # play badly with generate_novel_prefix() in DataTree, and will end up 

882 # generating very long tests with long runs of the rejection sample. 

883 if discard: 

884 self.consecutive_discard_counts[-1] += 1 

885 # 20 is a fairly arbitrary limit chosen mostly so that all of the 

886 # existing tests passed under it. Essentially no reasonable 

887 # generation should hit this limit when running in purely random 

888 # mode, but unreasonable generation is fairly widespread, and our 

889 # manipulation of the bitstream can make it more likely. 

890 if self.consecutive_discard_counts[-1] > 20: 

891 self.mark_invalid() 

892 else: 

893 self.consecutive_discard_counts[-1] = 0

905 def freeze(self): 

906 if self.frozen: 

907 assert isinstance(self.buffer, hbytes) 

908 return 

909 self.finish_time = benchmark_time() 

910 assert len(self.buffer) == self.index 

911 

912 # Always finish by closing all remaining examples so that we have a 

913 # valid tree. 

914 while self.depth >= 0: 

915 self.stop_example() 

916 

917 self.__example_record.freeze() 

918 

919 self.frozen = True 

920 

921 self.buffer = hbytes(self.buffer) 

922 self.events = frozenset(self.events)

926 def draw_bits(self, n, forced=None): 

927 """Return an ``n``-bit integer from the underlying source of 

928 bytes. If ``forced`` is set to an integer will instead 

929 ignore the underlying source and simulate a draw as if it had 

930 returned that integer.""" 

931 self.__assert_not_frozen("draw_bits") 

932 if n == 0: 

933 return 0 

934 assert n > 0 

935 n_bytes = bits_to_bytes(n) 

936 self.__check_capacity(n_bytes) 

937 

938 if forced is not None: 

939 buf = bytearray(int_to_bytes(forced, n_bytes)) 

940 else: 

941 buf = bytearray(self._draw_bytes(self, n_bytes)) 

942 assert len(buf) == n_bytes 

943 

944 # If we have a number of bits that is not a multiple of 8 

945 # we have to mask off the high bits. 

946 buf[0] &= BYTE_MASKS[n % 8] 

947 buf = hbytes(buf) 

948 result = int_from_bytes(buf) 

949 

950 self.observer.draw_bits(n, forced is not None, result) 

951 self.__example_record.draw_bits(n, forced) 

952 

953 initial = self.index 

954 

955 self.buffer.extend(buf) 

956 self.index = len(self.buffer) 

957 

958 if forced is not None: 

959 self.forced_indices.update(hrange(initial, self.index)) 

960 

961 self.blocks.add_endpoint(self.index) 

962 

963 assert bit_length(result) <= n 

964 return result
