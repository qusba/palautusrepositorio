308 def recursive_interpolate(key, value, section, backtrail): 

309 """The function that does the actual work. 

310 

311 ``value``: the string we're trying to interpolate. 

312 ``section``: the section in which that string was found 

313 ``backtrail``: a dict to keep track of where we've been, 

314 to detect and prevent infinite recursion loops 

315 

316 This is similar to a depth-first-search algorithm. 

317 """ 

318 # Have we been here already? 

319 if (key, section.name) in backtrail: 

320 # Yes - infinite loop detected 

321 raise InterpolationLoopError(key) 

322 # Place a marker on our backtrail so we won't come back here again 

323 backtrail[(key, section.name)] = 1 

324 

325 # Now start the actual work 

326 match = self._KEYCRE.search(value) 

327 while match: 

328 # The actual parsing of the match is implementation-dependent, 

329 # so delegate to our helper function 

330 k, v, s = self._parse_match(match) 

331 if k is None: 

332 # That's the signal that no further interpolation is needed 

333 replacement = v 

334 else: 

335 # Further interpolation may be needed to obtain final value 

336 replacement = recursive_interpolate(k, v, s, backtrail) 

337 # Replace the matched string with its final value 

338 start, end = match.span() 

339 value = ''.join((value[:start], replacement, value[end:])) 

340 new_search_start = start + len(replacement) 

341 # Pick up the next interpolation key, if any, for next time 

342 # through the while loop 

343 match = self._KEYCRE.search(value, new_search_start) 

344 

345 # Now safe to come back here again; remove marker from backtrail 

346 del backtrail[(key, section.name)] 

347 

348 return value 

349 

350 # Back in interpolate(), all we have to do is kick off the recursive 

351 # function with appropriate starting values 

352 value = recursive_interpolate(key, value, self.section, {}) 

353 return value

356 def _fetch(self, key): 

357 """Helper function to fetch values from owning section. 

358 

359 Returns a 2-tuple: the value, and the section where it was found. 

360 """ 

361 # switch off interpolation before we try and fetch anything ! 

362 save_interp = self.section.main.interpolation 

363 self.section.main.interpolation = False 

364 

365 # Start at section that "owns" this InterpolationEngine 

366 current_section = self.section 

367 while True: 

368 # try the current section first 

369 val = current_section.get(key) 

370 if val is not None and not isinstance(val, Section): 

371 break 

372 # try "DEFAULT" next 

373 val = current_section.get('DEFAULT', {}).get(key) 

374 if val is not None and not isinstance(val, Section): 

375 break 

376 # move up to parent and try again 

377 # top-level's parent is itself 

378 if current_section.parent is current_section: 

379 # reached top level, time to give up 

380 break 

381 current_section = current_section.parent 

382 

383 # restore interpolation to previous value before returning 

384 self.section.main.interpolation = save_interp 

385 if val is None: 

386 raise MissingInterpolationOption(key) 

387 return val, current_section

391 """Implementation-dependent helper function. 

392 

393 Will be passed a match object corresponding to the interpolation 

394 key we just found (e.g., "%(foo)s" or "$foo"). Should look up that 

395 key in the appropriate config file section (using the ``_fetch()`` 

396 helper function) and return a 3-tuple: (key, value, section) 

397 

398 ``key`` is the name of the key we're looking for 

399 ``value`` is the value found for that key 

400 ``section`` is a reference to the section where it was found 

401 

402 ``key`` and ``section`` should be None if no further 

403 interpolation should be performed on the resulting value 

404 (e.g., if we interpolated "$$" and returned "$"). 

405 """ 

406 raise NotImplementedError()

422class TemplateInterpolation(InterpolationEngine): 

423 """Behaves like string.Template.""" 

424 _cookie = '$' 

425 _delimiter = '$' 

426 _KEYCRE = re.compile(r""" 

427 \$(?: 

428 (?P<escaped>\$) | # Two $ signs 

429 (?P<named>[_a-z][_a-z0-9]*) | # $name format 

430 {(?P<braced>[^}]*)} # ${name} format 

431 ) 

432 """, re.IGNORECASE | re.VERBOSE) 

433 

434 def _parse_match(self, match): 

435 # Valid name (in or out of braces): fetch value from section 

436 key = match.group('named') or match.group('braced') 

437 if key is not None: 

438 value, section = self._fetch(key) 

439 return key, value, section 

440 # Escaped delimiter (e.g., $$): return single delimiter 

441 if match.group('escaped') is not None: 

442 # Return None for key and section to indicate it's time to stop 

443 return None, self._delimiter, None 

444 # Anything else: ignore completely, just return it unchanged 

445 return None, match.group(), None

1781

1792 # string type is for the python version we're dealing with 

1793 value = str(value)

1907 retval.append(cur_index)

1911 raise SyntaxError()

1919 else:

1928 raise SyntaxError()

2398 """

2400 (as returned by ``ConfigObj.validate``) into a flat list. 

2401

2406 ``results`` arguments - they are used by the function.)

2409  

2410 ([list of sections...], key, result) 

2411  

2412 If ``validate`` was called with ``preserve_errors=False`` (the default) 

2413 then ``result`` will always be ``False``. 

2414

2416 in. 

2417

2419 - or vice-versa) then key will be ``None``. 

2420  

2421 If the value (or section) was missing then ``result`` will be ``False``. 

2422

2424 was present, but failed the check, then ``result`` will be the exception 

2425 object returned. You can use this as a string that describes the failure. 

2426

2428 """ 

2429 if levels is None: 

2430 # first time called 

2431 levels = [] 

2432 results = [] 

2433 if res == True: 

2434 return sorted(results) 

2435 if res == False or isinstance(res, Exception):

2437 if levels: 

2438 levels.pop()

2453 #
