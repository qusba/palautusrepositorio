28def get_converter(from_unit, to_unit): 

29 """Like Unit._get_converter, except returns None if no scaling is needed, 

30 i.e., if the inferred scale is unity.""" 

31 try: 

32 scale = from_unit._to(to_unit) 

33 except UnitsError: 

34 return from_unit._apply_equivalencies( 

35 from_unit, to_unit, get_current_unit_registry().equivalencies) 

36 except AttributeError: 

37 raise UnitTypeError("Unit '{}' cannot be converted to '{}'" 

38 .format(from_unit, to_unit)) 

39 if scale == 1.: 

40 return None 

41 else: 

42 return lambda val: scale * val

45def get_converters_and_unit(f, unit1, unit2): 

46 converters = [None, None] 

47 # By default, we try adjusting unit2 to unit1, so that the result will 

48 # be unit1 as well. But if there is no second unit, we have to try 

49 # adjusting unit1 (to dimensionless, see below). 

50 if unit2 is None: 

51 if unit1 is None: 

52 # No units for any input -- e.g., np.add(a1, a2, out=q) 

53 return converters, dimensionless_unscaled 

54 changeable = 0 

55 # swap units. 

56 unit2 = unit1 

57 unit1 = None 

58 elif unit2 is unit1: 

59 # ensure identical units is fast ("==" is slow, so avoid that). 

60 return converters, unit1 

61 else: 

62 changeable = 1 

63 

64 # Try to get a converter from unit2 to unit1. 

65 if unit1 is None: 

66 try: 

67 converters[changeable] = get_converter(unit2, 

68 dimensionless_unscaled) 

69 except UnitsError: 

70 # special case: would be OK if unitless number is zero, inf, nan 

71 converters[1-changeable] = False 

72 return converters, unit2 

73 else: 

74 return converters, dimensionless_unscaled 

75 else: 

76 try: 

77 converters[changeable] = get_converter(unit2, unit1) 

78 except UnitsError: 

79 raise UnitConversionError( 

80 "Can only apply '{}' function to quantities " 

81 "with compatible dimensions" 

82 .format(f.__name__)) 

83 

84 return converters, unit1

299def helper_clip(f, unit1, unit2, unit3): 

300 # Treat the array being clipped as primary. 

301 converters = [None] 

302 if unit1 is None: 

303 result_unit = dimensionless_unscaled 

304 try: 

305 converters += [(None if unit is None else 

306 get_converter(unit, dimensionless_unscaled)) 

307 for unit in (unit2, unit3)] 

308 except UnitsError: 

309 raise UnitConversionError( 

310 "Can only apply '{}' function to quantities with " 

311 "compatible dimensions".format(f.__name__)) 

312 

313 else: 

314 result_unit = unit1 

315 for unit in unit2, unit3: 

316 try: 

317 converter = get_converter(_d(unit), result_unit) 

318 except UnitsError: 

319 if unit is None: 

320 # special case: OK if unitless number is zero, inf, nan 

321 converters.append(False) 

322 else: 

323 raise UnitConversionError( 

324 "Can only apply '{}' function to quantities with " 

325 "compatible dimensions".format(f.__name__)) 

326 else: 

327 converters.append(converter) 

328 

329 return converters, result_unit
