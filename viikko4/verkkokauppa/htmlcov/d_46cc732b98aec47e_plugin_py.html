243 class DocTestParserPlus(doctest.DocTestParser): 

244 """ 

245 An extension to the builtin DocTestParser that handles the 

246 special directives for skipping tests. 

247 

248 The directives are: 

249 

250 - ``.. doctest-skip::``: Skip the next doctest chunk. 

251 

252 - ``.. doctest-requires:: module1, module2``: Skip the next 

253 doctest chunk if the given modules/packages are not 

254 installed. 

255 

256 - ``.. doctest-skip-all``: Skip all subsequent doctests. 

257 """

259 def parse(self, s, name=None): 

260 result = doctest.DocTestParser.parse(self, s, name=name) 

261 

262 # result is a sequence of alternating text chunks and 

263 # doctest.Example objects. We need to look in the text 

264 # chunks for the special directives that help us determine 

265 # whether the following examples should be skipped. 

266 

267 required = [] 

268 skip_next = False 

269 skip_all = False 

270 

271 ext = os.path.splitext(name)[1] if name else '.rst' 

272 if ext not in comment_characters: 

273 warnings.warn("file format '{}' is not recognized, assuming " 

274 "'{}' as the comment character." 

275 .format(ext, comment_characters['rst'])) 

276 ext = '.rst' 

277 comment_char = comment_characters[ext] 

278 

279 ignore_warnings_context_needed = False 

280 

281 for entry in result: 

282 

283 if isinstance(entry, six.string_types) and entry: 

284 required = [] 

285 skip_next = False 

286 lines = entry.strip().splitlines() 

287 if any([re.match('{} doctest-skip-all'.format(comment_char), x.strip()) for x in lines]): 

288 skip_all = True 

289 continue 

290 

291 if not len(lines): 

292 continue 

293 

294 # We allow last and second to last lines to match to allow 

295 # special environment to be in between, e.g. \begin{python} 

296 last_lines = lines[-2:] 

297 matches = [re.match( 

298 r'{}\s+doctest-skip\s*::(\s+.*)?'.format(comment_char), 

299 last_line) for last_line in last_lines] 

300 

301 if len(matches) > 1: 

302 match = matches[0] or matches[1] 

303 else: 

304 match = matches[0] 

305 

306 if match: 

307 marker = match.group(1) 

308 if (marker is None or 

309 (marker.strip() == 'win32' and 

310 sys.platform == 'win32')): 

311 skip_next = True 

312 continue 

313 

314 matches = [re.match( 

315 r'{}\s+doctest-requires\s*::\s+(.*)'.format(comment_char), 

316 last_line) for last_line in last_lines] 

317 

318 if len(matches) > 1: 

319 match = matches[0] or matches[1] 

320 else: 

321 match = matches[0] 

322 

323 if match: 

324 # 'a a' or 'a,a' or 'a, a'-> [a, a] 

325 required = re.split(r'\s*[,\s]\s*', match.group(1)) 

326 elif isinstance(entry, doctest.Example): 

327 

328 # If warnings are to be ignored we need to catch them by 

329 # wrapping the source in a context manager. 

330 if entry.options.get(IGNORE_WARNINGS, False): 

331 entry.source = ("with _doctestplus_ignore_all_warnings():\n" 

332 + indent(entry.source, ' ')) 

333 ignore_warnings_context_needed = True 

334 

335 has_required_modules = DocTestFinderPlus.check_required_modules(required) 

336 if skip_all or skip_next or not has_required_modules: 

337 entry.options[doctest.SKIP] = True 

338 

339 if config.getoption('remote_data', 'none') != 'any' and entry.options.get(REMOTE_DATA): 

340 entry.options[doctest.SKIP] = True 

341 

342 # We insert the definition of the context manager to ignore 

343 # warnings at the start of the file if needed. 

344 if ignore_warnings_context_needed: 

345 result.insert(0, doctest.Example(source=IGNORE_WARNINGS_CONTEXT, want='')) 

346 

347 return result

416 def pytest_collect_file(self, path, parent): 

417 """Implements an enhanced version of the doctest module from py.test 

418 (specifically, as enabled by the --doctest-modules option) which 

419 supports skipping all doctests in a specific docstring by way of a 

420 special ``__doctest_skip__`` module-level variable. It can also skip 

421 tests that have special requirements by way of 

422 ``__doctest_requires__``. 

423 

424 ``__doctest_skip__`` should be a list of functions, classes, or class 

425 methods whose docstrings should be ignored when collecting doctests. 

426 

427 This also supports wildcard patterns. For example, to run doctests in 

428 a class's docstring, but skip all doctests in its modules use, at the 

429 module level:: 

430 

431 __doctest_skip__ = ['ClassName.*'] 

432 

433 You may also use the string ``'.'`` in ``__doctest_skip__`` to refer 

434 to the module itself, in case its module-level docstring contains 

435 doctests. 

436 

437 ``__doctest_requires__`` should be a dictionary mapping wildcard 

438 patterns (in the same format as ``__doctest_skip__``) to a list of one 

439 or more modules that should be *importable* in order for the tests to 

440 run. For example, if some tests require the scipy module to work they 

441 will be skipped unless ``import scipy`` is possible. It is also 

442 possible to use a tuple of wildcard patterns as a key in this dict:: 

443 

444 __doctest_requires__ = {('func1', 'func2'): ['scipy']} 

445 

446 """ 

447 for ignore_path in self._ignore_paths: 

448 if ignore_path.common(path) == ignore_path: 

449 return None 

450 

451 if path.ext == '.py': 

452 if path.basename == 'conf.py': 

453 return None 

454 

455 # Don't override the built-in doctest plugin 

456 return self._doctest_module_item_cls(path, parent) 

457 elif any([path.check(fnmatch=pat) for pat in self._file_globs]): 

458 # Ignore generated .rst files 

459 parts = str(path).split(os.path.sep) 

460 

461 # Don't test files that start with a _ 

462 if path.basename.startswith('_'): 

463 return None 

464 

465 # Don't test files in directories that start with a '_' if those 

466 # directories are inside docs. Note that we *should* allow for 

467 # example /tmp/_q/docs/file.rst but not /tmp/docs/_build/file.rst 

468 # If we don't find 'docs' in the path, we should just skip this 

469 # check to be safe. We also want to skip any api sub-directory 

470 # of docs. 

471 if 'docs' in parts: 

472 # We index from the end on the off chance that the temporary 

473 # directory includes 'docs' in the path, e.g. 

474 # /tmp/docs/371j/docs/index.rst You laugh, but who knows! :) 

475 # Also, it turns out lists don't have an rindex method. Huh??!! 

476 docs_index = len(parts) - 1 - parts[::-1].index('docs') 

477 if any(x.startswith('_') or x == 'api' for x in parts[docs_index:]): 

478 return None 

479 

480 # TODO: Get better names on these items when they are 

481 # displayed in py.test output 

482 return self._doctest_textfile_item_cls(path, parent)

485class DocTestFinderPlus(doctest.DocTestFinder): 

486 """Extension to the default `doctest.DoctestFinder` that supports 

487 ``__doctest_skip__`` magic. See `pytest_collect_file` for more details. 

488 """ 

489 

490 # Caches the results of import attempts 

491 _import_cache = {} 

492 _module_checker = ModuleChecker() 

493 

494 @classmethod 

495 def check_required_modules(cls, mods): 

496 """Check that modules in `mods` list are available. 

497 

498 Parameters 

499 ---------- 

500 mods : list of str 

501 List of modules. Modules can have specified versions (eg 'numpy>=1.15') 

502 

503 Returns 

504 ------- 

505 bool 

506 True if all modules in list are available. 

507 """ 

508 for mod in mods: 

509 if mod in cls._import_cache: 

510 if not cls._import_cache[mod]: 

511 return False 

512 

513 if cls._module_checker.check(mod): 

514 cls._import_cache[mod] = True 

515 else: 

516 cls._import_cache[mod] = False 

517 return False 

518 return True

520 def find(self, obj, name=None, module=None, globs=None, extraglobs=None): 

521 tests = doctest.DocTestFinder.find(self, obj, name, module, globs, extraglobs) 

522 if hasattr(obj, '__doctest_skip__') or hasattr(obj, '__doctest_requires__'): 

523 if name is None and hasattr(obj, '__name__'): 

524 name = obj.__name__ 

525 else: 

526 raise ValueError("DocTestFinder.find: name must be given " 

527 "when obj.__name__ doesn't exist: {!r}" 

528 .format((type(obj),))) 

529 

530 def test_filter(test): 

531 for pat in getattr(obj, '__doctest_skip__', []): 

532 if pat == '*': 

533 return False 

534 elif pat == '.' and test.name == name: 

535 return False 

536 elif fnmatch.fnmatch(test.name, '.'.join((name, pat))): 

537 return False 

538 

539 reqs = getattr(obj, '__doctest_requires__', {}) 

540 for pats, mods in six.iteritems(reqs): 

541 if not isinstance(pats, tuple): 

542 pats = (pats,) 

543 for pat in pats: 

544 if not fnmatch.fnmatch(test.name, '.'.join((name, pat))): 

545 continue 

546 if not self.check_required_modules(mods): 

547 return False 

548 return True 

549 

550 tests = list(filter(test_filter, tests)) 

551 

552 return tests
