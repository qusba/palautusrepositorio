73@attr.s(slots=True) 

74class TreeNode(object): 

75 """Node in a tree that corresponds to previous interactions with 

76 a ``ConjectureData`` object according to some fixed test function. 

77 

78 This is functionally a variant patricia trie. 

79 See https://en.wikipedia.org/wiki/Radix_tree for the general idea, 

80 but what this means in particular here is that we have a very deep 

81 but very lightly branching tree and rather than store this as a fully 

82 recursive structure we flatten prefixes and long branches into 

83 lists. This significantly compacts the storage requirements. 

84 

85 A single ``TreeNode`` corresponds to a previously seen sequence 

86 of calls to ``ConjectureData`` which we have never seen branch, 

87 followed by a ``transition`` which describes what happens next. 

88 """ 

89 

90 # Records the previous sequence of calls to ``data.draw_bits``, 

91 # with the ``n_bits`` argument going in ``bit_lengths`` and the 

92 # values seen in ``values``. These should always have the same 

93 # length. 

94 bit_lengths = attr.ib(default=attr.Factory(IntList)) 

95 values = attr.ib(default=attr.Factory(IntList)) 

96 

97 # The indices of of the calls to ``draw_bits`` that we have stored 

98 # where ``forced`` is not None. Stored as None if no indices 

99 # have been forced, purely for space saving reasons (we force 

100 # quite rarely). 

101 __forced = attr.ib(default=None, init=False) 

102 

103 # What happens next after observing this sequence of calls. 

104 # Either: 

105 # 

106 # * ``None``, indicating we don't know yet. 

107 # * A ``Branch`` object indicating that there is a ``draw_bits`` 

108 # call that we have seen take multiple outcomes there. 

109 # * A ``Conclusion`` object indicating that ``conclude_test`` 

110 # was called here. 

111 transition = attr.ib(default=None) 

112 

113 # A tree node is exhausted if every possible sequence of 

114 # draws below it has been explored. We store this information 

115 # on a field and update it when performing operations that 

116 # could change the answer. 

117 # 

118 # A node may start exhausted, e.g. because it it leads 

119 # immediately to a conclusion, but can only go from 

120 # non-exhausted to exhausted when one of its children 

121 # becomes exhausted or it is marked as a conclusion. 

122 # 

123 # Therefore we only need to check whether we need to update 

124 # this field when the node is first created in ``split_at`` 

125 # or when we have walked a path through this node to a 

126 # conclusion in ``TreeRecordingObserver``. 

127 is_exhausted = attr.ib(default=False, init=False) 

128 

129 @property 

130 def forced(self): 

131 if not self.__forced: 

132 return EMPTY 

133 return self.__forced 

134 

135 def mark_forced(self, i): 

136 """Note that the value at index ``i`` was forced.""" 

137 assert 0 <= i < len(self.values) 

138 if self.__forced is None: 

139 self.__forced = set() 

140 self.__forced.add(i) 

141 

142 def split_at(self, i): 

143 """Splits the tree so that it can incorporate 

144 a decision at the ``draw_bits`` call corresponding 

145 to position ``i``, or raises ``Flaky`` if that was 

146 meant to be a forced node.""" 

147 

148 if i in self.forced: 

149 inconsistent_generation() 

150 

151 assert not self.is_exhausted 

152 

153 key = self.values[i] 

154 

155 child = TreeNode( 

156 bit_lengths=self.bit_lengths[i + 1 :], 

157 values=self.values[i + 1 :], 

158 transition=self.transition, 

159 ) 

160 self.transition = Branch(bit_length=self.bit_lengths[i], children={key: child}) 

161 if self.__forced is not None: 

162 child.__forced = {j - i - 1 for j in self.__forced if j > i} 

163 self.__forced = {j for j in self.__forced if j < i} 

164 child.check_exhausted() 

165 del self.values[i:] 

166 del self.bit_lengths[i:] 

167 assert len(self.values) == len(self.bit_lengths) == i 

168 

169 def check_exhausted(self): 

170 """Recalculates ``self.is_exhausted`` if necessary then returns 

171 it.""" 

172 if ( 

173 not self.is_exhausted 

174 and len(self.forced) == len(self.values) 

175 and self.transition is not None 

176 ): 

177 if isinstance(self.transition, Conclusion): 

178 self.is_exhausted = True 

179 elif len(self.transition.children) == self.transition.max_children: 

180 self.is_exhausted = all( 

181 v.is_exhausted for v in self.transition.children.values() 

182 ) 

183 return self.is_exhausted

186class DataTree(object): 

187 """Tracks the tree structure of a collection of ConjectureData 

188 objects, for use in ConjectureRunner.""" 

189 

190 def __init__(self): 

191 self.root = TreeNode() 

192 

193 @property 

194 def is_exhausted(self): 

195 """Returns True if every possible node is dead and thus the language 

196 described must have been fully explored.""" 

197 return self.root.is_exhausted 

198 

199 def generate_novel_prefix(self, random): 

200 """Generate a short random string that (after rewriting) is not 

201 a prefix of any buffer previously added to the tree. 

202 

203 The resulting prefix is essentially arbitrary - it would be nice 

204 for it to be uniform at random, but previous attempts to do that 

205 have proven too expensive. 

206 """ 

207 assert not self.is_exhausted 

208 novel_prefix = bytearray() 

209 

210 def append_int(n_bits, value): 

211 novel_prefix.extend(int_to_bytes(value, bits_to_bytes(n_bits))) 

212 

213 current_node = self.root 

214 while True: 

215 assert not current_node.is_exhausted 

216 for i, (n_bits, value) in enumerate( 

217 zip(current_node.bit_lengths, current_node.values) 

218 ): 

219 if i in current_node.forced: 

220 append_int(n_bits, value) 

221 else: 

222 while True: 

223 k = random.getrandbits(n_bits) 

224 if k != value: 

225 append_int(n_bits, k) 

226 break 

227 # We've now found a value that is allowed to 

228 # vary, so what follows is not fixed. 

229 return hbytes(novel_prefix) 

230 else: 

231 assert not isinstance(current_node.transition, Conclusion) 

232 if current_node.transition is None: 

233 return hbytes(novel_prefix) 

234 branch = current_node.transition 

235 assert isinstance(branch, Branch) 

236 n_bits = branch.bit_length 

237 

238 while True: 

239 k = random.getrandbits(n_bits) 

240 try: 

241 child = branch.children[k] 

242 except KeyError: 

243 append_int(n_bits, k) 

244 return hbytes(novel_prefix) 

245 if not child.is_exhausted: 

246 append_int(n_bits, k) 

247 current_node = child 

248 break

250 def rewrite(self, buffer): 

251 """Use previously seen ConjectureData objects to return a tuple of 

252 the rewritten buffer and the status we would get from running that 

253 buffer with the test function. If the status cannot be predicted 

254 from the existing values it will be None.""" 

255 buffer = hbytes(buffer) 

256 

257 data = ConjectureData.for_buffer(buffer) 

258 try: 

259 self.simulate_test_function(data) 

260 return (data.buffer, data.status) 

261 except PreviouslyUnseenBehaviour: 

262 return (buffer, None)

264 def simulate_test_function(self, data): 

265 """Run a simulated version of the test function recorded by 

266 this tree. Note that this does not currently call ``stop_example`` 

267 or ``start_example`` as these are not currently recorded in the 

268 tree. This will likely change in future.""" 

269 node = self.root 

270 try: 

271 while True: 

272 for i, (n_bits, previous) in enumerate( 

273 zip(node.bit_lengths, node.values) 

274 ): 

275 v = data.draw_bits( 

276 n_bits, forced=node.values[i] if i in node.forced else None 

277 ) 

278 if v != previous: 

279 raise PreviouslyUnseenBehaviour() 

280 if isinstance(node.transition, Conclusion): 

281 t = node.transition 

282 data.conclude_test(t.status, t.interesting_origin) 

283 elif node.transition is None: 

284 raise PreviouslyUnseenBehaviour() 

285 else: 

286 v = data.draw_bits(node.transition.bit_length) 

287 try: 

288 node = node.transition.children[v] 

289 except KeyError: 

290 raise PreviouslyUnseenBehaviour() 

291 except StopTest: 

292 pass

298class TreeRecordingObserver(DataObserver): 

299 def __init__(self, tree): 

300 self.__current_node = tree.root 

301 self.__index_in_current_node = 0 

302 self.__trail = [self.__current_node] 

303 

304 def draw_bits(self, n_bits, forced, value): 

305 i = self.__index_in_current_node 

306 self.__index_in_current_node += 1 

307 node = self.__current_node 

308 assert len(node.bit_lengths) == len(node.values) 

309 if i < len(node.bit_lengths): 

310 if n_bits != node.bit_lengths[i]: 

311 inconsistent_generation() 

312 # Note that we don't check whether a previously 

313 # forced value is now free. That will be caught 

314 # if we ever split the node there, but otherwise 

315 # may pass silently. This is acceptable because it 

316 # means we skip a hash set lookup on every 

317 # draw and that's a pretty niche failure mode. 

318 if forced and i not in node.forced: 

319 inconsistent_generation() 

320 if value != node.values[i]: 

321 node.split_at(i) 

322 assert i == len(node.values) 

323 new_node = TreeNode() 

324 branch = node.transition 

325 branch.children[value] = new_node 

326 self.__current_node = new_node 

327 self.__index_in_current_node = 0 

328 else: 

329 trans = node.transition 

330 if trans is None: 

331 node.bit_lengths.append(n_bits) 

332 node.values.append(value) 

333 if forced: 

334 node.mark_forced(i) 

335 elif isinstance(trans, Conclusion): 

336 assert trans.status != Status.OVERRUN 

337 # We tried to draw where history says we should have 

338 # stopped 

339 inconsistent_generation() 

340 else: 

341 assert isinstance(trans, Branch) 

342 if n_bits != trans.bit_length: 

343 inconsistent_generation() 

344 try: 

345 self.__current_node = trans.children[value] 

346 except KeyError: 

347 self.__current_node = trans.children.setdefault(value, TreeNode()) 

348 self.__index_in_current_node = 0 

349 if self.__trail[-1] is not self.__current_node: 

350 self.__trail.append(self.__current_node) 

351 

352 def conclude_test(self, status, interesting_origin): 

353 """Says that ``status`` occurred at node ``node``. This updates the 

354 node if necessary and checks for consistency.""" 

355 if status == Status.OVERRUN: 

356 return 

357 i = self.__index_in_current_node 

358 node = self.__current_node 

359 

360 if i < len(node.values) or isinstance(node.transition, Branch): 

361 inconsistent_generation() 

362 

363 new_transition = conclusion(status, interesting_origin) 

364 

365 if node.transition is not None and node.transition != new_transition: 

366 # As an, I'm afraid, horrible bodge, we deliberately ignore flakiness 

367 # where tests go from interesting to valid, because it's much easier 

368 # to produce good error messages for these further up the stack. 

369 if ( 

370 node.transition.status != Status.INTERESTING 

371 or new_transition.status != Status.VALID 

372 ): 

373 raise Flaky( 

374 "Inconsistent test results! Test case was %r on first run but %r on second" 

375 % (node.transition, new_transition) 

376 ) 

377 else: 

378 node.transition = new_transition 

379 

380 assert node is self.__trail[-1] 

381 node.check_exhausted() 

382 assert len(node.values) > 0 or node.check_exhausted() 

383 

384 for t in reversed(self.__trail): 

385 # Any node we've traversed might have now become exhausted. 

386 # We check from the right. As soon as we hit a node that 

387 # isn't exhausted, this automatically implies that all of 

388 # its parents are not exhausted, so we stop. 

389 if not t.check_exhausted(): 

390 break
